40  /**
41   * A callable task that will process a given set of NVD CVE xml files and update
42   * the Cve Database accordingly.
43   *
44   * @author Jeremy Long
45   */
46  @ThreadSafe
47  public class ProcessTask implements Callable<ProcessTask> {
48  
49      /**
50       * The logger.
51       */
52      private static final Logger LOGGER = LoggerFactory.getLogger(ProcessTask.class);
53      /**
54       * A field to store any update exceptions that occur during the "call".
55       */
56      private UpdateException exception = null;
57      /**
58       * A reference to the CveDB.
59       */
60      private final CveDB cveDB;
61      /**
62       * A reference to the callable download task.
63       */
64      private final DownloadTask filePair;
65      /**
66       * A reference to the properties.
67       */
68      private final DatabaseProperties properties;
69      /**
70       * A reference to the global settings object.
71       */
72      private final Settings settings;
73  
74      /**
75       * Get the value of exception.
76       *
77       * @return the value of exception
78       */
79      public UpdateException getException() {
80          return exception;
81      }
82  
83      /**
84       * Set the value of exception.
85       *
86       * @param exception new value of exception
87       */
88      public void setException(UpdateException exception) {
89          this.exception = exception;
90      }
91  
92      /**
93       * Constructs a new ProcessTask used to process an NVD CVE update.
94       *
95       * @param cveDB the data store object
96       * @param filePair the download task that contains the URL references to
97       * download
98       * @param settings a reference to the global settings object; this is
99       * necessary so that when the thread is started the dependencies have a
100      * correct reference to the global settings.
101      */
102     public ProcessTask(final CveDB cveDB, final DownloadTask filePair, Settings settings) {
103         this.cveDB = cveDB;
104         this.filePair = filePair;
105         this.properties = cveDB.getDatabaseProperties();
106         this.settings = settings;
107     }
108 
109     /**
110      * Implements the callable interface.
111      *
112      * @return this object
113      * @throws Exception thrown if there is an exception; note that any
114      * UpdateExceptions are simply added to the tasks exception collection
115      */
116     @Override
117     public ProcessTask call() throws Exception {
118         try {
119             processFiles();
120         } catch (UpdateException ex) {
121             this.exception = ex;
122         } finally {
123             settings.cleanup(false);
124         }
125         return this;
126     }
127 
128     /**
129      * Imports the NVD CVE XML File into the Lucene Index.
130      *
131      * @param file the file containing the NVD CVE XML
132      * @param oldVersion contains the file containing the NVD CVE XML 1.2
133      * @throws ParserConfigurationException is thrown if there is a parser
134      * configuration exception
135      * @throws SAXException is thrown if there is a SAXException
136      * @throws IOException is thrown if there is a IO Exception
137      * @throws SQLException is thrown if there is a SQL exception
138      * @throws DatabaseException is thrown if there is a database exception
139      * @throws ClassNotFoundException thrown if the h2 database driver cannot be
140      * loaded
141      */
142     protected void importXML(File file, File oldVersion) throws ParserConfigurationException,
143             SAXException, IOException, SQLException, DatabaseException, ClassNotFoundException {
144 
145         final SAXParser saxParser = XmlUtils.buildSecureSaxParser();
146 
147         final NvdCve12Handler cve12Handler = new NvdCve12Handler();
148         saxParser.parse(oldVersion, cve12Handler);
149         final Map<String, List<VulnerableSoftware>> prevVersionVulnMap = cve12Handler.getVulnerabilities();
150 
151         final NvdCve20Handler cve20Handler = new NvdCve20Handler();
152         cve20Handler.setCveDB(cveDB);
153         cve20Handler.setPrevVersionVulnMap(prevVersionVulnMap);
154         saxParser.parse(file, cve20Handler);
155     }
156 
157     /**
158      * Processes the NVD CVE XML file and imports the data into the DB.
159      *
160      * @throws UpdateException thrown if there is an error loading the data into
161      * the database
162      */
163     private void processFiles() throws UpdateException {
164         LOGGER.info("Processing Started for NVD CVE - {}", filePair.getNvdCveInfo().getId());
165         final long startProcessing = System.currentTimeMillis();
166         try {
167             importXML(filePair.getFirst(), filePair.getSecond());
168             cveDB.commit();
169             properties.save(filePair.getNvdCveInfo());
170         } catch (ParserConfigurationException | SAXException | SQLException | DatabaseException | ClassNotFoundException | IOException ex) {
171             throw new UpdateException(ex);
172         } finally {
173             filePair.cleanup();
174         }
175         LOGGER.info("Processing Complete for NVD CVE - {}  ({} ms)", filePair.getNvdCveInfo().getId(),
176                 System.currentTimeMillis() - startProcessing);
177     }
178 }
