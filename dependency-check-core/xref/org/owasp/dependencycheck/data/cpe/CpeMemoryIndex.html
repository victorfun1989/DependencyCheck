50  /**
51   * <p>
52   * An in memory Lucene index that contains the vendor/product combinations from
53   * the CPE (application) identifiers within the NVD CVE data.</p>
54   *
55   * This is the last remaining singleton in dependency-check-core; The use of
56   * this singleton - while it may not technically be thread-safe (one database
57   * used to build this index may not have the same entries as another) the risk
58   * of this is currently believed to be small. As this memory index consumes a
59   * large amount of memory we will remain using the singleton pattern for now.
60   *
61   * @author Jeremy Long
62   */
63  @ThreadSafe
64  public final class CpeMemoryIndex implements AutoCloseable {
65  
66      /**
67       * Singleton instance.
68       */
69      private static final CpeMemoryIndex INSTANCE = new CpeMemoryIndex();
70      /**
71       * The logger.
72       */
73      private static final Logger LOGGER = LoggerFactory.getLogger(CpeMemoryIndex.class);
74      /**
75       * The in memory Lucene index.
76       */
77      private RAMDirectory index;
78      /**
79       * The Lucene IndexReader.
80       */
81      private IndexReader indexReader;
82      /**
83       * The Lucene IndexSearcher.
84       */
85      private IndexSearcher indexSearcher;
86      /**
87       * The Lucene Analyzer used for Searching.
88       */
89      private Analyzer searchingAnalyzer;
90      /**
91       * The Lucene QueryParser used for Searching.
92       */
93      private QueryParser queryParser;
94      /**
95       * The product field analyzer.
96       */
97      private SearchFieldAnalyzer productFieldAnalyzer;
98      /**
99       * The vendor field analyzer.
100      */
101     private SearchFieldAnalyzer vendorFieldAnalyzer;
102     /**
103      * Track the number of current users of the Lucene index; used to track it
104      * it is okay to actually close the index.
105      */
106     private final AtomicInteger usageCount = new AtomicInteger(0);
107 
108     /**
109      * private constructor for singleton.
110      */
111     private CpeMemoryIndex() {
112     }
113 
114     /**
115      * Gets the singleton instance of the CpeMemoryIndex.
116      *
117      * @return the instance of the CpeMemoryIndex
118      */
119     public static CpeMemoryIndex getInstance() {
120         return INSTANCE;
121     }
122 
123     /**
124      * Creates and loads data into an in memory index.
125      *
126      * @param cve the data source to retrieve the cpe data
127      * @throws IndexException thrown if there is an error creating the index
128      */
129     public synchronized void open(CveDB cve) throws IndexException {
130         if (INSTANCE.usageCount.addAndGet(1) == 1) {
131             index = new RAMDirectory();
132             buildIndex(cve);
133             try {
134                 indexReader = DirectoryReader.open(index);
135             } catch (IOException ex) {
136                 throw new IndexException(ex);
137             }
138             indexSearcher = new IndexSearcher(indexReader);
139             searchingAnalyzer = createSearchingAnalyzer();
140             queryParser = new QueryParser(Fields.DOCUMENT_KEY, searchingAnalyzer);
141         }
142     }
143 
144     /**
145      * returns whether or not the index is open.
146      *
147      * @return whether or not the index is open
148      */
149     public synchronized boolean isOpen() {
150         return INSTANCE.usageCount.get() > 0;
151     }
152 
153     /**
154      * Creates an Analyzer for searching the CPE Index.
155      *
156      * @return the CPE Analyzer.
157      */
158     private Analyzer createSearchingAnalyzer() {
159         final Map<String, Analyzer> fieldAnalyzers = new HashMap<>();
160         fieldAnalyzers.put(Fields.DOCUMENT_KEY, new KeywordAnalyzer());
161         productFieldAnalyzer = new SearchFieldAnalyzer();
162         vendorFieldAnalyzer = new SearchFieldAnalyzer();
163         fieldAnalyzers.put(Fields.PRODUCT, productFieldAnalyzer);
164         fieldAnalyzers.put(Fields.VENDOR, vendorFieldAnalyzer);
165 
166         return new PerFieldAnalyzerWrapper(new KeywordAnalyzer(), fieldAnalyzers);
167     }
168 
169     /**
170      * Closes the CPE Index.
171      */
172     @Override
173     public synchronized void close() {
174         final int count = INSTANCE.usageCount.get() - 1;
175         if (count <= 0) {
176             INSTANCE.usageCount.set(0);
177             if (searchingAnalyzer != null) {
178                 searchingAnalyzer.close();
179                 searchingAnalyzer = null;
180             }
181             if (indexReader != null) {
182                 try {
183                     indexReader.close();
184                 } catch (IOException ex) {
185                     LOGGER.trace("", ex);
186                 }
187                 indexReader = null;
188             }
189             queryParser = null;
190             indexSearcher = null;
191             if (index != null) {
192                 index.close();
193                 index = null;
194             }
195         }
196     }
197 
198     /**
199      * Builds the CPE Lucene Index based off of the data within the CveDB.
200      *
201      * @param cve the data base containing the CPE data
202      * @throws IndexException thrown if there is an issue creating the index
203      */
204     private void buildIndex(CveDB cve) throws IndexException {
205         try (Analyzer analyzer = createSearchingAnalyzer();
206                 IndexWriter indexWriter = new IndexWriter(index, new IndexWriterConfig(analyzer))) {
207             // Tip: reuse the Document and Fields for performance...
208             // See "Re-use Document and Field instances" from
209             // http://wiki.apache.org/lucene-java/ImproveIndexingSpeed
210             final Document doc = new Document();
211             final Field v = new TextField(Fields.VENDOR, Fields.VENDOR, Field.Store.YES);
212             final Field p = new TextField(Fields.PRODUCT, Fields.PRODUCT, Field.Store.YES);
213             doc.add(v);
214             doc.add(p);
215 
216             final Set<Pair<String, String>> data = cve.getVendorProductList();
217             for (Pair<String, String> pair : data) {
218                 if (pair.getLeft() != null && pair.getRight() != null) {
219                     v.setStringValue(pair.getLeft());
220                     p.setStringValue(pair.getRight());
221                     indexWriter.addDocument(doc);
222                     productFieldAnalyzer.reset();
223                     vendorFieldAnalyzer.reset();
224                 }
225             }
226             indexWriter.commit();
227         } catch (DatabaseException ex) {
228             LOGGER.debug("", ex);
229             throw new IndexException("Error reading CPE data", ex);
230         } catch (IOException ex) {
231             throw new IndexException("Unable to close an in-memory index", ex);
232         }
233     }
234 
235     /**
236      * Searches the index using the given search string.
237      *
238      * @param searchString the query text
239      * @param maxQueryResults the maximum number of documents to return
240      * @return the TopDocs found by the search
241      * @throws ParseException thrown when the searchString is invalid
242      * @throws IOException is thrown if there is an issue with the underlying
243      * Index
244      */
245     public synchronized TopDocs search(String searchString, int maxQueryResults) throws ParseException, IOException {
246         final Query query = parseQuery(searchString);
247         productFieldAnalyzer.reset();
248         vendorFieldAnalyzer.reset();
249         return search(query, maxQueryResults);
250     }
251 
252     /**
253      * Parses the given string into a Lucene Query.
254      *
255      * @param searchString the search text
256      * @return the Query object
257      * @throws ParseException thrown if the search text cannot be parsed
258      */
259     protected Query parseQuery(String searchString) throws ParseException {
260         if (searchString == null || searchString.trim().isEmpty()) {
261             throw new ParseException("Query is null or empty");
262         }
263         LOGGER.debug(searchString);
264         final Query query = queryParser.parse(searchString);
265         return query;
266     }
267 
268     /**
269      * Searches the index using the given query.
270      *
271      * @param query the query used to search the index
272      * @param maxQueryResults the max number of results to return
273      * @return the TopDocs found be the query
274      * @throws CorruptIndexException thrown if the Index is corrupt
275      * @throws IOException thrown if there is an IOException
276      */
277     public synchronized TopDocs search(Query query, int maxQueryResults) throws CorruptIndexException, IOException {
278         return indexSearcher.search(query, maxQueryResults);
279     }
280 
281     /**
282      * Retrieves a document from the Index.
283      *
284      * @param documentId the id of the document to retrieve
285      * @return the Document
286      * @throws IOException thrown if there is an IOException
287      */
288     public synchronized Document getDocument(int documentId) throws IOException {
289         return indexSearcher.doc(documentId);
290     }
291 
292     /**
293      * Returns the number of CPE entries stored in the index.
294      *
295      * @return the number of CPE entries stored in the index
296      */
297     public synchronized int numDocs() {
298         if (indexReader == null) {
299             return -1;
300         }
301         return indexReader.numDocs();
302     }
