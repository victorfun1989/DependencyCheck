81       * {@link Engine} execution modes.

83      public enum Mode {
84          /**
85           * In evidence collection mode the {@link Engine} only collects evidence
86           * from the scan targets, and doesn't require a database.
87           */
88          EVIDENCE_COLLECTION(
89                  false,
90                  INITIAL,
91                  PRE_INFORMATION_COLLECTION,
92                  INFORMATION_COLLECTION,
93                  POST_INFORMATION_COLLECTION
94          ),
95          /**
96           * In evidence processing mode the {@link Engine} processes the evidence
97           * collected using the {@link #EVIDENCE_COLLECTION} mode. Dependencies
98           * should be injected into the {@link Engine} using
99           * {@link Engine#setDependencies(List)}.
100          */
101         EVIDENCE_PROCESSING(
102                 true,
103                 PRE_IDENTIFIER_ANALYSIS,
104                 IDENTIFIER_ANALYSIS,
105                 POST_IDENTIFIER_ANALYSIS,
106                 PRE_FINDING_ANALYSIS,
107                 FINDING_ANALYSIS,
108                 POST_FINDING_ANALYSIS,
109                 FINAL
110         ),
111         /**
112          * In standalone mode the {@link Engine} will collect and process
113          * evidence in a single execution.

115         STANDALONE(true, AnalysisPhase.values());

265     /**
266      * Loads the analyzers specified in the configuration file (or system
267      * properties).
268      */
269     private void loadAnalyzers() {
270         if (!analyzers.isEmpty()) {
271             return;
272         }
273         for (AnalysisPhase phase : mode.getPhases()) {
274             analyzers.put(phase, new ArrayList<Analyzer>());
275         }
276         final AnalyzerService service = new AnalyzerService(serviceClassLoader, settings);
277         final List<Analyzer> iterator = service.getAnalyzers(mode.getPhases());
278         for (Analyzer a : iterator) {
279             a.initialize(this.settings);
280             analyzers.get(a.getAnalysisPhase()).add(a);
281             if (a instanceof FileTypeAnalyzer) {
282                 this.fileTypeAnalyzers.add((FileTypeAnalyzer) a);
283             }
284         }
285     }

575     /**
576      * Scans a specified file. If a dependency is identified it is added to the
577      * dependency collection.
578      *
579      * @param file The file to scan
580      * @param projectReference the name of the project or scope in which the
581      * dependency was identified
582      * @return the scanned dependency
583      * @since v1.4.4
584      */
585     protected synchronized Dependency scanFile(File file, String projectReference) {
586         Dependency dependency = null;
587         if (file.isFile()) {
588             if (accept(file)) {
589                 dependency = new Dependency(file);
590                 if (projectReference != null) {
591                     dependency.addProjectReference(projectReference);
592                 }
593                 final String sha1 = dependency.getSha1sum();
594                 boolean found = false;
595 
596                 if (sha1 != null) {
597                     for (Dependency existing : dependencies) {
598                         if (sha1.equals(existing.getSha1sum())) {
599                             found = true;
600                             if (projectReference != null) {
601                                 existing.addProjectReference(projectReference);
602                             }
603                             if (existing.getActualFilePath() != null && dependency.getActualFilePath() != null
604                                     && !existing.getActualFilePath().equals(dependency.getActualFilePath())) {
605                                 existing.addRelatedDependency(dependency);
606                             } else {
607                                 dependency = existing;
608                             }
609                             break;
610                         }
611                     }
612                 }
613                 if (!found) {
614                     dependencies.add(dependency);
615                     dependenciesExternalView = null;
616                 }
617             }
618         } else {
619             LOGGER.debug("Path passed to scanFile(File) is not a file that can be scanned by dependency-check: {}. Skipping the file.", file);
620         }
621         return dependency;
622     }

624     /**
625      * Runs the analyzers against all of the dependencies. Since the mutable
626      * dependencies list is exposed via {@link #getDependencies()}, this method
627      * iterates over a copy of the dependencies list. Thus, the potential for
628      * {@link java.util.ConcurrentModificationException}s is avoided, and
629      * analyzers may safely add or remove entries from the dependencies list.
630      * <p>
631      * Every effort is made to complete analysis on the dependencies. In some
632      * cases an exception will occur with part of the analysis being performed
633      * which may not affect the entire analysis. If an exception occurs it will
634      * be included in the thrown exception collection.
635      *
636      * @throws ExceptionCollection a collections of any exceptions that occurred
637      * during analysis
638      */
639     public void analyzeDependencies() throws ExceptionCollection {
640         final List<Throwable> exceptions = Collections.synchronizedList(new ArrayList<Throwable>());
641 
642         initializeAndUpdateDatabase(exceptions);
643 
644         //need to ensure that data exists
645         try {
646             ensureDataExists();
647         } catch (NoDataException ex) {
648             throwFatalExceptionCollection("Unable to continue dependency-check analysis.", ex, exceptions);
649         }
650 
651         LOGGER.debug("\n----------------------------------------------------\nBEGIN ANALYSIS\n----------------------------------------------------");
652         LOGGER.info("Analysis Started");
653         final long analysisStart = System.currentTimeMillis();
654 
655         // analysis phases
656         for (AnalysisPhase phase : mode.getPhases()) {
657             final List<Analyzer> analyzerList = analyzers.get(phase);
658 
659             for (final Analyzer analyzer : analyzerList) {
660                 final long analyzerStart = System.currentTimeMillis();
661                 try {
662                     initializeAnalyzer(analyzer);
663                 } catch (InitializationException ex) {
664                     exceptions.add(ex);
665                     if (ex.isFatal()) {
666                         continue;
667                     }
668                 }
669 
670                 if (analyzer.isEnabled()) {
671                     executeAnalysisTasks(analyzer, exceptions);
672 
673                     final long analyzerDurationMillis = System.currentTimeMillis() - analyzerStart;
674                     final long analyzerDurationSeconds = TimeUnit.MILLISECONDS.toSeconds(analyzerDurationMillis);
675                     LOGGER.info("Finished {} ({} seconds)", analyzer.getName(), analyzerDurationSeconds);
676                 } else {
677                     LOGGER.debug("Skipping {} (not enabled)", analyzer.getName());
678                 }
679             }
680         }
681         for (AnalysisPhase phase : mode.getPhases()) {
682             final List<Analyzer> analyzerList = analyzers.get(phase);
683 
684             for (Analyzer a : analyzerList) {
685                 closeAnalyzer(a);
686             }
687         }
688 
689         LOGGER.debug("\n----------------------------------------------------\nEND ANALYSIS\n----------------------------------------------------");
690         final long analysisDurationSeconds = TimeUnit.MILLISECONDS.toSeconds(System.currentTimeMillis() - analysisStart);
691         LOGGER.info("Analysis Complete ({} seconds)", analysisDurationSeconds);
692         if (exceptions.size() > 0) {
693             throw new ExceptionCollection("One or more exceptions occurred during dependency-check analysis", exceptions);
694         }
695     }

697     /**
698      * Performs any necessary updates and initializes the database.
699      *
700      * @param exceptions a collection to store non-fatal exceptions
701      * @throws ExceptionCollection thrown if fatal exceptions occur
702      */
703     private void initializeAndUpdateDatabase(final List<Throwable> exceptions) throws ExceptionCollection {
704         if (!mode.isDatabaseRequired()) {
705             return;
706         }
707         boolean autoUpdate = true;
708         try {
709             autoUpdate = settings.getBoolean(Settings.KEYS.AUTO_UPDATE);
710         } catch (InvalidSettingException ex) {
711             LOGGER.debug("Invalid setting for auto-update; using true.");
712             exceptions.add(ex);
713         }
714         if (autoUpdate) {
715             try {
716                 doUpdates(true);
717             } catch (UpdateException ex) {
718                 exceptions.add(ex);
719                 LOGGER.warn("Unable to update Cached Web DataSource, using local "
720                         + "data instead. Results may not include recent vulnerabilities.");
721                 LOGGER.debug("Update Error", ex);
722             } catch (DatabaseException ex) {
723                 final String msg;
724                 if (ex.getMessage().contains("Unable to connect") && ConnectionFactory.isH2Connection(settings)) {
725                     msg = "Unable to update connect to the database - if this error persists it may be "
726                             + "due to a corrupt database. Consider running `purge` to delete the existing database";
727                 } else {
728                     msg = "Unable to connect to the database";
729                 }
730                 throw new ExceptionCollection("Unable to connect to the database", ex);
731             }
732         } else {
733             try {
734                 if (ConnectionFactory.isH2Connection(settings) && !ConnectionFactory.h2DataFileExists(settings)) {
735                     throw new ExceptionCollection(new NoDataException("Autoupdate is disabled and the database does not exist"), true);
736                 } else {
737                     openDatabase(true, true);
738                 }
739             } catch (IOException ex) {
740                 throw new ExceptionCollection(new DatabaseException("Autoupdate is disabled and unable to connect to the database"), true);
741             } catch (DatabaseException ex) {
742                 throwFatalExceptionCollection("Unable to connect to the dependency-check database.", ex, exceptions);
743             }
744         }
745     }

749      *
750      * @param exceptions a collection of exceptions that occurred during
751      * analysis
752      * @param analyzer the analyzer to execute
753      * @throws ExceptionCollection thrown if exceptions occurred during analysis
754      */
755     protected void executeAnalysisTasks(Analyzer analyzer, List<Throwable> exceptions) throws ExceptionCollection {
756         LOGGER.debug("Starting {}", analyzer.getName());
757         final List<AnalysisTask> analysisTasks = getAnalysisTasks(analyzer, exceptions);
758         final ExecutorService executorService = getExecutorService(analyzer);
759 
760         try {
761             final int timeout = settings.getInt(Settings.KEYS.ANALYSIS_TIMEOUT, 20);
762             final List<Future<Void>> results = executorService.invokeAll(analysisTasks, timeout, TimeUnit.MINUTES);
763 
764             // ensure there was no exception during execution
765             for (Future<Void> result : results) {
766                 try {
767                     result.get();
768                 } catch (ExecutionException e) {
769                     throwFatalExceptionCollection("Analysis task failed with a fatal exception.", e, exceptions);
770                 } catch (CancellationException e) {
771                     throwFatalExceptionCollection("Analysis task was cancelled.", e, exceptions);
772                 }
773             }
774         } catch (InterruptedException e) {
775             Thread.currentThread().interrupt();
776             throwFatalExceptionCollection("Analysis has been interrupted.", e, exceptions);
777         } finally {
778             executorService.shutdown();
779         }
780     }

798     /**
799      * Returns the executor service for a given analyzer.
800      *
801      * @param analyzer the analyzer to obtain an executor
802      * @return the executor service
803      */
804     protected ExecutorService getExecutorService(Analyzer analyzer) {
805         if (analyzer.supportsParallelProcessing()) {
806             final int maximumNumberOfThreads = Runtime.getRuntime().availableProcessors();
807             LOGGER.debug("Parallel processing with up to {} threads: {}.", maximumNumberOfThreads, analyzer.getName());
808             return Executors.newFixedThreadPool(maximumNumberOfThreads);
809         } else {
810             LOGGER.debug("Parallel processing is not supported: {}.", analyzer.getName());
811             return Executors.newSingleThreadExecutor();
812         }
813     }

874     /**
875      * Cycles through the cached web data sources and calls update on all of
876      * them.
877      *
878      * @param remainOpen whether or not the database connection should remain
879      * open
880      * @throws UpdateException thrown if the operation fails
881      * @throws DatabaseException if the operation fails due to a local database failure
882      */
883     public void doUpdates(boolean remainOpen) throws UpdateException, DatabaseException {
884         if (mode.isDatabaseRequired()) {
885             H2DBLock dblock = null;
886             try {
887                 if (ConnectionFactory.isH2Connection(settings)) {
888                     dblock = new H2DBLock(settings);
889                     LOGGER.debug("locking for update");
890                     dblock.lock();
891                 }
892                 openDatabase(false, false);
893                 LOGGER.info("Checking for updates");
894                 final long updateStart = System.currentTimeMillis();
895                 final UpdateService service = new UpdateService(serviceClassLoader);
896                 final Iterator<CachedWebDataSource> iterator = service.getDataSources();
897                 while (iterator.hasNext()) {
898                     final CachedWebDataSource source = iterator.next();
899                     source.update(this);
900                 }
901                 database.close();
902                 database = null;
903                 LOGGER.info("Check for updates complete ({} ms)", System.currentTimeMillis() - updateStart);
904                 if (remainOpen) {
905                     openDatabase(true, false);
906                 }
907             } catch (H2DBLockException ex) {
908                 throw new UpdateException("Unable to obtain an exclusive lock on the H2 database to perform updates", ex);
909             } finally {
910                 if (dblock != null) {
911                     dblock.release();
912                 }
913             }
914         } else {
915             LOGGER.info("Skipping update check in evidence collection mode.");
916         }
917     }

933     /**
934      * <p>
935      * This method is only public for unit/integration testing. This method
936      * should not be called by any integration that uses
937      * dependency-check-core.</p>
938      * <p>
939      * Opens the database connection; if readOnly is true a copy of the database
940      * will be made.</p>
941      *
942      * @param readOnly whether or not the database connection should be readonly
943      * @param lockRequired whether or not a lock needs to be acquired when
944      * opening the database
945      * @throws DatabaseException if the database connection could not be created
946      */
947     public void openDatabase(boolean readOnly, boolean lockRequired) throws DatabaseException {
948         if (mode.isDatabaseRequired() && database == null) {
949             //needed to update schema any required schema changes
950             database = new CveDB(settings);
951             if (readOnly
952                     && ConnectionFactory.isH2Connection(settings)
953                     && settings.getString(Settings.KEYS.DB_CONNECTION_STRING).contains("file:%s")) {
954                 H2DBLock lock = null;
955                 try {
956                     final File db = ConnectionFactory.getH2DataFile(settings);
957                     if (db.isFile()) {
958                         database.close();
959                         if (lockRequired) {
960                             lock = new H2DBLock(settings);
961                             lock.lock();
962                         }
963                         LOGGER.debug("copying database");
964                         final File temp = settings.getTempDirectory();
965                         final File tempDB = new File(temp, db.getName());
966                         Files.copy(db.toPath(), tempDB.toPath());
967                         LOGGER.debug("copying complete '{}'", temp.toPath());
968                         settings.setString(Settings.KEYS.H2_DATA_DIRECTORY, temp.getPath());
969                         final String connStr = settings.getString(Settings.KEYS.DB_CONNECTION_STRING);
970                         if (!connStr.contains("ACCESS_MODE_DATA")) {
971                             settings.setString(Settings.KEYS.DB_CONNECTION_STRING, connStr + "ACCESS_MODE_DATA=r");
972                         }
973                         database = new CveDB(settings);
974                     }
975                 } catch (IOException ex) {
976                     throw new DatabaseException("Unable to open db in read only mode", ex);
977                 } catch (H2DBLockException ex) {
978                     throw new DatabaseException("Failed to obtain lock - unable to open db in read only mode", ex);
979                 } finally {
980                     if (lock != null) {
981                         lock.release();
982                     }
983                 }
984             }
985         }
986     }

1013      * Checks all analyzers to see if an extension is supported.
1014      *
1015      * @param file a file extension
1016      * @return true or false depending on whether or not the file extension is
1017      * supported
1018      */
1019     @Override
1020     public boolean accept(File file) {
1021         if (file == null) {
1022             return false;
1023         }
1024         boolean scan = false;
1025         for (FileTypeAnalyzer a : this.fileTypeAnalyzers) {
1026             /* note, we can't break early on this loop as the analyzers need to know if
1027              they have files to work on prior to initialization */
1028             scan |= a.accept(file);
1029         }
1030         return scan;
1031     }
