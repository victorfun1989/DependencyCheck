49  /**
50   * Used to analyze Node Package Manager (npm) package.json files, and collect
51   * information that can be used to determine the associated CPE.

53   * @author Dale Visser
54   */
55  @ThreadSafe

58      /**
59       * The logger.
60       */
61      private static final Logger LOGGER = LoggerFactory.getLogger(NodePackageAnalyzer.class);
62      /**
63       * A descriptor for the type of dependencies processed or added by this
64       * analyzer.
65       */
66      public static final String DEPENDENCY_ECOSYSTEM = NPM_DEPENDENCY_ECOSYSTEM;
67      /**
68       * The name of the analyzer.

70      private static final String ANALYZER_NAME = "Node.js Package Analyzer";
71      /**
72       * The phase that this analyzer is intended to run in.
73       */
74      private static final AnalysisPhase ANALYSIS_PHASE = AnalysisPhase.INFORMATION_COLLECTION;
75      /**
76       * The file name to scan.
77       */
78      public static final String PACKAGE_JSON = "package.json";
79      /**
80       * The file name to scan.
81       */
82      public static final String PACKAGE_LOCK_JSON = "package-lock.json";
83      /**
84       * The file name to scan.
85       */
86      public static final String SHRINKWRAP_JSON = "npm-shrinkwrap.json";
87      /**
88       * Filter that detects files named "package-lock.json" or
89       * "npm-shrinkwrap.json".
90       */
91      private static final FileFilter PACKAGE_JSON_FILTER = FileFilterBuilder.newInstance()
92              .addFilenames(PACKAGE_LOCK_JSON, SHRINKWRAP_JSON).build();

104     /**
105      * Performs validation on the configuration to ensure that the correct
106      * analyzers are in place.
107      *
108      * @param engine the dependency-check engine
109      * @throws InitializationException thrown if there is a configuration error
110      */
111     @Override
112     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
113         if (engine.getMode() != Mode.EVIDENCE_COLLECTION) {
114             try {
115                 final Settings settings = engine.getSettings();
116                 final String[] tmp = settings.getArray(Settings.KEYS.ECOSYSTEM_SKIP_CPEANALYZER);
117                 if (tmp != null) {
118                     final List<String> skipEcosystems = Arrays.asList(tmp);
119                     if (skipEcosystems.contains(DEPENDENCY_ECOSYSTEM)
120                             && !settings.getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED)) {
121                         LOGGER.debug("NodePackageAnalyzer enabled without a corresponding vulnerability analyzer");
122                         final String msg = "Invalid Configuration: enabling the Node Package Analyzer without "
123                                 + "using the Node Audit Analyzer is not supported.";
124                         throw new InitializationException(msg);
125                     } else if (!skipEcosystems.contains(DEPENDENCY_ECOSYSTEM)) {
126                         LOGGER.warn("Using the CPE Analyzer with Node.js can result in many false positives.");
127                     }
128                 }
129             } catch (InvalidSettingException ex) {
130                 throw new InitializationException("Unable to read configuration settings", ex);
131             }
132         }
133     }

166     /**
167      * Determines if the Node Audit analyzer is enabled.
168      *
169      * @param engine a reference to the dependency-check engine
170      * @return <code>true</code> if the Node Audit Analyzer is enabled;
171      * otherwise <code>false</code>
172      */
173     private boolean isNodeAuditEnabled(Engine engine) {
174         for (Analyzer a : engine.getAnalyzers()) {
175             if (a instanceof NodeAuditAnalyzer) {
176                 return a.isEnabled();
177             }
178         }
179         return false;
180     }

182     @Override
183     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
184         if (isNodeAuditEnabled(engine) && !PACKAGE_LOCK_JSON.equals(dependency.getFileName())) {
185             engine.removeDependency(dependency);
186         }
187         final File dependencyFile = dependency.getActualFile();
188         if (!dependencyFile.isFile() || dependencyFile.length() == 0 || !shouldProcess(dependencyFile)) {
189             return;
190         }
191         final File baseDir = dependencyFile.getParentFile();
192         if (PACKAGE_JSON.equals(dependency.getFileName())) {
193             final File lockfile = new File(baseDir, PACKAGE_LOCK_JSON);
194             final File shrinkwrap = new File(baseDir, SHRINKWRAP_JSON);
195             if (shrinkwrap.exists() || lockfile.exists()) {
196                 return;
197             }
198         } else if (PACKAGE_LOCK_JSON.equals(dependency.getFileName())) {
199             final File shrinkwrap = new File(baseDir, SHRINKWRAP_JSON);
200             if (shrinkwrap.exists()) {
201                 return;
202             }
203         }
204         final File nodeModules = new File(baseDir, "node_modules");
205         if (!nodeModules.isDirectory()) {
206             LOGGER.warn("Analyzing `{}` - however, the node_modules directory does not exist. "
207                     + "Please run `npm install` prior to running dependency-check", dependencyFile.toString());
208             return;
209         }
210 
211         try (JsonReader jsonReader = Json.createReader(FileUtils.openInputStream(dependencyFile))) {
212             final JsonObject json = jsonReader.readObject();
213             final String parentName = json.getString("name", "");
214             final String parentVersion = json.getString("version", "");
215             if (parentName.isEmpty()) {
216                 return;
217             }
218             final String parentPackage;
219             if (!parentVersion.isEmpty()) {
220                 parentPackage = String.format("%s:%s", parentName, parentVersion);
221             } else {
222                 parentPackage = parentName;
223             }
224             processDependencies(json, baseDir, dependencyFile, parentPackage, engine);
225         } catch (JsonException e) {
226             LOGGER.warn("Failed to parse package.json file.", e);
227         } catch (IOException e) {
228             throw new AnalysisException("Problem occurred while reading dependency file.", e);
229         }
230     }
231 
232     /**
233      * Process the dependencies in the lock file by first parsing its
234      * dependencies and then finding the package.json for the module and adding
235      * it as a dependency.
236      *
237      * @param json the data to process
238      * @param baseDir the base directory being scanned
239      * @param rootFile the root package-lock/npm-shrinkwrap being analyzed
240      * @param parentPackage the parent package name of the current node
241      * @param engine a reference to the dependency-check engine
242      * @throws AnalysisException thrown if there is an exception
243      */
244     private void processDependencies(JsonObject json, File baseDir, File rootFile,
245             String parentPackage, Engine engine) throws AnalysisException {
246         if (json.containsKey("dependencies")) {
247             final JsonObject deps = json.getJsonObject("dependencies");
248             for (Map.Entry<String, JsonValue> entry : deps.entrySet()) {
249                 final JsonObject jo = (JsonObject) entry.getValue();
250                 final String name = entry.getKey();
251                 final String version = jo.getString("version");
252                 final File base = Paths.get(baseDir.getPath(), "node_modules", name).toFile();
253                 final File f = new File(base, PACKAGE_JSON);
254 
255                 if (jo.containsKey("dependencies")) {
256                     final String subPackageName = String.format("%s/%s:%s", parentPackage, name, version);
257                     processDependencies(jo, base, rootFile, subPackageName, engine);
258                 }
259 
260                 final Dependency child;
261                 if (f.exists()) {
262                     //TOOD - we should use the integrity value instead of calculating the SHA1/MD5
263                     child = new Dependency(f);
264                     child.setEcosystem(DEPENDENCY_ECOSYSTEM);
265 
266                     try (JsonReader jr = Json.createReader(FileUtils.openInputStream(f))) {
267                         final JsonObject childJson = jr.readObject();
268                         gatherEvidence(childJson, child);
269 
270                     } catch (JsonException e) {
271                         LOGGER.warn("Failed to parse package.json file from dependency.", e);
272                     } catch (IOException e) {
273                         throw new AnalysisException("Problem occurred while reading dependency file.", e);
274                     }
275                 } else {
276                     LOGGER.warn("Unable to find node module: {}", f.toString());
277                     child = new Dependency(rootFile, true);
278                     child.setEcosystem(DEPENDENCY_ECOSYSTEM);
279                     //TOOD - we should use the integrity value instead of calculating the SHA1/MD5
280                     child.setSha1sum(Checksum.getSHA1Checksum(String.format("%s:%s", name, version)));
281                     child.setSha256sum(Checksum.getSHA256Checksum(String.format("%s:%s", name, version)));
282                     child.setMd5sum(Checksum.getMD5Checksum(String.format("%s:%s", name, version)));
283                     child.addEvidence(EvidenceType.VENDOR, rootFile.getName(), "name", name, Confidence.HIGHEST);
284                     child.addEvidence(EvidenceType.PRODUCT, rootFile.getName(), "name", name, Confidence.HIGHEST);
285                     child.addEvidence(EvidenceType.VERSION, rootFile.getName(), "version", version, Confidence.HIGHEST);
286                     child.setName(name);
287                     child.setVersion(version);
288                     final String packagePath = String.format("%s:%s", name, version);
289                     child.setDisplayFileName(packagePath);
290                     child.setPackagePath(packagePath);
291                 }
292 
293                 child.addProjectReference(parentPackage);
294                 child.setEcosystem(DEPENDENCY_ECOSYSTEM);
295 
296                 final Dependency existing = findDependency(engine, name, version);
297                 if (existing != null) {
298                     if (existing.isVirtual()) {
299                         DependencyMergingAnalyzer.mergeDependencies(child, existing, null);
300                         engine.removeDependency(existing);
301                         engine.addDependency(child);
302                     } else {
303                         DependencyBundlingAnalyzer.mergeDependencies(existing, child, null);
304                     }
305                 } else {
306                     engine.addDependency(child);
307                 }
308             }
309         }
310     }
