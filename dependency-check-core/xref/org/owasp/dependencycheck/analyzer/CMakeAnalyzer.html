41  /**
42   * <p>
43   * Used to analyze CMake build files, and collect information that can be used
44   * to determine the associated CPE.</p>
45   * <p>
46   * Note: This analyzer catches straightforward invocations of the project
47   * command, plus some other observed patterns of version inclusion in real CMake
48   * projects. Many projects make use of older versions of CMake and/or use custom
49   * "homebrew" ways to insert version information. Hopefully as the newer CMake
50   * call pattern grows in usage, this analyzer allow more CPEs to be
51   * identified.</p>
52   *
53   * @author Dale Visser
54   */
55  @Experimental
56  public class CMakeAnalyzer extends AbstractFileTypeAnalyzer {
57  
58      /**
59       * A descriptor for the type of dependencies processed or added by this
60       * analyzer.
61       */
62      public static final String DEPENDENCY_ECOSYSTEM = "CMAKE";
63  
64      /**
65       * The logger.
66       */
67      private static final Logger LOGGER = LoggerFactory.getLogger(CMakeAnalyzer.class);
68  
69      /**
70       * Used when compiling file scanning regex patterns.
71       */
72      private static final int REGEX_OPTIONS = Pattern.DOTALL | Pattern.CASE_INSENSITIVE | Pattern.MULTILINE;
73  
74      /**
75       * Regex to extract the product information.
76       */
77      private static final Pattern PROJECT = Pattern.compile(
78              "^ *project *\\([ \\n]*(\\w+)[ \\n]*.*?\\)", REGEX_OPTIONS);
79  
80      /**
81       * Regex to extract product and version information.
82       *
83       * Group 1: Product
84       *
85       * Group 2: Version
86       */
87      private static final Pattern SET_VERSION = Pattern.compile(
88              "^ *set\\s*\\(\\s*(\\w+)_version\\s+\"?(\\d+(?:\\.\\d+)+)[\\s\"]?\\)", REGEX_OPTIONS);
89  
90      /**
91       * Detects files that can be analyzed.
92       */
93      private static final FileFilter FILTER = FileFilterBuilder.newInstance().addExtensions(".cmake")
94              .addFilenames("CMakeLists.txt").build();
95  
96      /**
97       * Returns the name of the CMake analyzer.
98       *
99       * @return the name of the analyzer
100      */
101     @Override
102     public String getName() {
103         return "CMake Analyzer";
104     }
105 
106     /**
107      * Tell that we are used for information collection.
108      *
109      * @return INFORMATION_COLLECTION
110      */
111     @Override
112     public AnalysisPhase getAnalysisPhase() {
113         return AnalysisPhase.INFORMATION_COLLECTION;
114     }

116     /**
117      * Returns the set of supported file extensions.
118      *
119      * @return the set of supported file extensions
120      */
121     @Override
122     protected FileFilter getFileFilter() {
123         return FILTER;
124     }

126     /**
127      * Initializes the analyzer.
128      *
129      * @param engine a reference to the dependency-check engine
130      * @throws InitializationException thrown if an exception occurs getting an
131      * instance of SHA1
132      */
133     @Override
134     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
135         //do nothing
136     }

138     /**
139      * Analyzes python packages and adds evidence to the dependency.
140      *
141      * @param dependency the dependency being analyzed
142      * @param engine the engine being used to perform the scan
143      * @throws AnalysisException thrown if there is an unrecoverable error
144      * analyzing the dependency
145      */
146     @Override
147     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
148         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
149         final File file = dependency.getActualFile();
150         final String name = file.getName();
151         final String contents;
152         try {
153             contents = FileUtils.readFileToString(file, Charset.defaultCharset()).trim();
154         } catch (IOException e) {
155             throw new AnalysisException(
156                     "Problem occurred while reading dependency file.", e);
157         }
158         if (StringUtils.isNotBlank(contents)) {
159             final Matcher m = PROJECT.matcher(contents);
160             int count = 0;
161             while (m.find()) {
162                 count++;
163                 LOGGER.debug(String.format(
164                         "Found project command match with %d groups: %s",
165                         m.groupCount(), m.group(0)));
166                 final String group = m.group(1);
167                 LOGGER.debug("Group 1: {}", group);
168                 dependency.addEvidence(EvidenceType.PRODUCT, name, "Project", group, Confidence.HIGH);
169                 dependency.addEvidence(EvidenceType.VENDOR, name, "Project", group, Confidence.HIGH);
170                 dependency.setName(group);
171             }
172             LOGGER.debug("Found {} matches.", count);
173             analyzeSetVersionCommand(dependency, engine, contents);
174         }
175     }
176 
177     /**
178      * Extracts the version information from the contents. If more then one
179      * version is found additional dependencies are added to the dependency
180      * list.
181      *
182      * @param dependency the dependency being analyzed
183      * @param engine the dependency-check engine
184      * @param contents the version information
185      */
186     private void analyzeSetVersionCommand(Dependency dependency, Engine engine, String contents) {
187         Dependency currentDep = dependency;
188 
189         final Matcher m = SET_VERSION.matcher(contents);
190         int count = 0;
191         while (m.find()) {
192             count++;
193             LOGGER.debug("Found project command match with {} groups: {}",
194                     m.groupCount(), m.group(0));
195             String product = m.group(1);
196             final String version = m.group(2);
197             LOGGER.debug("Group 1: {}", product);
198             LOGGER.debug("Group 2: {}", version);
199             final String aliasPrefix = "ALIASOF_";
200             if (product.startsWith(aliasPrefix)) {
201                 product = product.replaceFirst(aliasPrefix, "");
202             }
203             if (count > 1) {
204                 //TODO - refactor so we do not assign to the parameter (checkstyle)
205                 currentDep = new Dependency(dependency.getActualFile());
206                 currentDep.setEcosystem(DEPENDENCY_ECOSYSTEM);
207                 final String filePath = String.format("%s:%s", dependency.getFilePath(), product);
208                 currentDep.setFilePath(filePath);
209 
210                 currentDep.setSha1sum(Checksum.getSHA1Checksum(filePath));
211                 currentDep.setSha256sum(Checksum.getSHA256Checksum(filePath));
212                 currentDep.setMd5sum(Checksum.getMD5Checksum(filePath));
213                 engine.addDependency(currentDep);
214             }
215             final String source = currentDep.getFileName();
216             currentDep.addEvidence(EvidenceType.PRODUCT, source, "Product", product, Confidence.MEDIUM);
217             currentDep.addEvidence(EvidenceType.VENDOR, source, "Vendor", product, Confidence.MEDIUM);
218             currentDep.addEvidence(EvidenceType.VERSION, source, "Version", version, Confidence.MEDIUM);
219             currentDep.setName(product);
220             currentDep.setVersion(version);
221         }
222         LOGGER.debug("Found {} matches.", count);
223     }
224 
225     @Override
226     protected String getAnalyzerEnabledSettingKey() {
227         return Settings.KEYS.ANALYZER_CMAKE_ENABLED;
228     }
