93      /**
94       * Builds the beginnings of a List for ProcessBuilder
95       *
96       * @return the list of arguments to begin populating the ProcessBuilder
97       */
98      protected List<String> buildArgumentList() {
99          // Use file.separator as a wild guess as to whether this is Windows
100         final List<String> args = new ArrayList<>();
101         if (!SystemUtils.IS_OS_WINDOWS) {
102             if (getSettings().getString(Settings.KEYS.ANALYZER_ASSEMBLY_MONO_PATH) != null) {
103                 args.add(getSettings().getString(Settings.KEYS.ANALYZER_ASSEMBLY_MONO_PATH));
104             } else if (isInPath("mono")) {
105                 args.add("mono");
106             } else {
107                 return null;
108             }
109         }
110         args.add(grokAssemblyExe.getPath());
111         return args;
112     }

114     /**
115      * Performs the analysis on a single Dependency.
116      *
117      * @param dependency the dependency to analyze
118      * @param engine the engine to perform the analysis under
119      * @throws AnalysisException if anything goes sideways
120      */
121     @Override
122     public void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
123         final File test = new File(dependency.getActualFilePath());
124         if (!test.isFile()) {
125             throw new AnalysisException(String.format("%s does not exist and cannot be analyzed by dependency-check",
126                     dependency.getActualFilePath()));
127         }
128         if (grokAssemblyExe == null) {
129             LOGGER.warn("GrokAssembly didn't get deployed");
130             return;
131         }
132         final List<String> args = buildArgumentList();
133         if (args == null) {
134             LOGGER.warn("Assembly Analyzer was unable to execute");
135             return;
136         }
137         args.add(dependency.getActualFilePath());
138         final ProcessBuilder pb = new ProcessBuilder(args);
139         Document doc = null;
140         try {
141             final Process proc = pb.start();
142             final DocumentBuilder builder = XmlUtils.buildSecureDocumentBuilder();
143 
144             doc = builder.parse(proc.getInputStream());
145 
146             // Try evacuating the error stream
147             final String errorStream = IOUtils.toString(proc.getErrorStream(), StandardCharsets.UTF_8);
148             if (null != errorStream && !errorStream.isEmpty()) {
149                 LOGGER.warn("Error from GrokAssembly: {}", errorStream);
150             }
151 
152             int rc = 0;
153             try {
154                 rc = proc.waitFor();
155             } catch (InterruptedException ie) {
156                 Thread.currentThread().interrupt();
157                 return;
158             }
159             if (rc == 3) {
160                 LOGGER.debug("{} is not a .NET assembly or executable and as such cannot be analyzed by dependency-check",
161                         dependency.getActualFilePath());
162                 return;
163             } else if (rc != 0) {
164                 LOGGER.debug("Return code {} from GrokAssembly; dependency-check is unable to analyze the library: {}",
165                         rc, dependency.getActualFilePath());
166                 return;
167             }
168 
169             final XPath xpath = XPathFactory.newInstance().newXPath();
170 
171             // First, see if there was an error
172             final String error = xpath.evaluate("/assembly/error", doc);
173             if (error != null && !error.isEmpty()) {
174                 throw new AnalysisException(error);
175             }
176 
177             final String version = xpath.evaluate("/assembly/version", doc);
178             if (version != null) {
179                 dependency.addEvidence(EvidenceType.VERSION, "grokassembly", "version", version, Confidence.HIGHEST);
180             }
181 
182             final String vendor = xpath.evaluate("/assembly/company", doc);
183             if (vendor != null) {
184                 dependency.addEvidence(EvidenceType.VENDOR, "grokassembly", "vendor", vendor, Confidence.HIGH);
185             }
186 
187             final String product = xpath.evaluate("/assembly/product", doc);
188             if (product != null) {
189                 dependency.addEvidence(EvidenceType.PRODUCT, "grokassembly", "product", product, Confidence.HIGH);
190             }
191 
192         } catch (ParserConfigurationException pce) {
193             throw new AnalysisException("Error initializing the assembly analyzer", pce);
194         } catch (IOException | XPathExpressionException ioe) {
195             throw new AnalysisException(ioe);
196         } catch (SAXException saxe) {
197             LOGGER.error("----------------------------------------------------");
198             LOGGER.error("Failed to read the Assembly Analyzer results. "
199                     + "On some systems mono-runtime and mono-devel need to be installed.");
200             LOGGER.error("----------------------------------------------------");
201             throw new AnalysisException("Couldn't parse Assembly Analyzer results (GrokAssembly)", saxe);
202         }
203     }

205     /**
206      * Initialize the analyzer. In this case, extract GrokAssembly.exe to a
207      * temporary location.
208      *
209      * @param engine a reference to the dependency-check engine
210      * @throws InitializationException thrown if anything goes wrong
211      */
212     @Override
213     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
214         final File tempFile;
215         final File cfgFile;
216         try {
217             tempFile = File.createTempFile("GKA", ".exe", getSettings().getTempDirectory());
218             cfgFile = new File(tempFile.getPath() + ".config");
219         } catch (IOException ex) {
220             setEnabled(false);
221             throw new InitializationException("Unable to create temporary file for the assembly analyzer", ex);
222         }
223         try (FileOutputStream fos = new FileOutputStream(tempFile);
224                 InputStream is = FileUtils.getResourceAsStream("GrokAssembly.exe");
225                 FileOutputStream fosCfg = new FileOutputStream(cfgFile);
226                 InputStream isCfg = FileUtils.getResourceAsStream("GrokAssembly.exe.config")) {
227             IOUtils.copy(is, fos);
228             grokAssemblyExe = tempFile;
229             LOGGER.debug("Extracted GrokAssembly.exe to {}", grokAssemblyExe.getPath());
230             IOUtils.copy(isCfg, fosCfg);
231             grokAssemblyConfig = cfgFile;
232             LOGGER.debug("Extracted GrokAssembly.exe.config to {}", cfgFile);
233         } catch (IOException ioe) {
234             this.setEnabled(false);
235             LOGGER.warn("Could not extract GrokAssembly.exe: {}", ioe.getMessage());
236             throw new InitializationException("Could not extract GrokAssembly.exe", ioe);
237         }
238 
239         // Now, need to see if GrokAssembly actually runs from this location.
240         final List<String> args = buildArgumentList();
241         //TODO this creates an "unreported" error - if someone doesn't look
242         // at the command output this could easily be missed (especially in an
243         // Ant or Maven build.
244         //
245         // We need to create a non-fatal warning error type that will
246         // get added to the report.
247         //TODO this idea needs to get replicated to the bundle audit analyzer.
248         if (args == null) {
249             setEnabled(false);
250             LOGGER.error("----------------------------------------------------");
251             LOGGER.error(".NET Assembly Analyzer could not be initialized and at least one "
252                     + "'exe' or 'dll' was scanned. The 'mono' executable could not be found on "
253                     + "the path; either disable the Assembly Analyzer or configure the path mono. "
254                     + "On some systems mono-runtime and mono-devel need to be installed.");
255             LOGGER.error("----------------------------------------------------");
256             return;
257         }
258         try {
259             final ProcessBuilder pb = new ProcessBuilder(args);
260             final Process p = pb.start();
261             // Try evacuating the error stream
262             IOUtils.copy(p.getErrorStream(), NullOutputStream.NULL_OUTPUT_STREAM);
263 
264             final DocumentBuilder builder = XmlUtils.buildSecureDocumentBuilder();
265             final Document doc = builder.parse(p.getInputStream());
266             final XPath xpath = XPathFactory.newInstance().newXPath();
267             final String error = xpath.evaluate("/assembly/error", doc);
268             if (p.waitFor() != 1 || error == null || error.isEmpty()) {
269                 LOGGER.warn("An error occurred with the .NET AssemblyAnalyzer, please see the log for more details.");
270                 LOGGER.debug("GrokAssembly.exe is not working properly");
271                 grokAssemblyExe = null;
272                 setEnabled(false);
273                 throw new InitializationException("Could not execute .NET AssemblyAnalyzer");
274             }
275         } catch (InitializationException e) {
276             setEnabled(false);
277             throw e;
278         } catch (IOException | ParserConfigurationException | SAXException | XPathExpressionException | InterruptedException e) {
279             LOGGER.warn("An error occurred with the .NET AssemblyAnalyzer;\n"
280                     + "this can be ignored unless you are scanning .NET DLLs. Please see the log for more details.");
281             LOGGER.debug("Could not execute GrokAssembly {}", e.getMessage());
282             setEnabled(false);
283             throw new InitializationException("An error occurred with the .NET AssemblyAnalyzer", e);
284         }
285     }

287     /**
288      * Removes resources used from the local file system.
289      *
290      * @throws Exception thrown if there is a problem closing the analyzer
291      */
292     @Override
293     public void closeAnalyzer() throws Exception {
294         try {
295             if (grokAssemblyExe != null && !grokAssemblyExe.delete()) {
296                 LOGGER.debug("Unable to delete temporary GrokAssembly.exe; attempting delete on exit");
297                 grokAssemblyExe.deleteOnExit();
298             }
299         } catch (SecurityException se) {
300             LOGGER.debug("Can't delete temporary GrokAssembly.exe");
301             grokAssemblyExe.deleteOnExit();
302         }
303         try {
304             if (grokAssemblyConfig != null && !grokAssemblyConfig.delete()) {
305                 LOGGER.debug("Unable to delete temporary GrokAssembly.exe.config; attempting delete on exit");
306                 grokAssemblyConfig.deleteOnExit();
307             }
308         } catch (SecurityException se) {
309             LOGGER.debug("Can't delete temporary GrokAssembly.exe.config");
310             grokAssemblyConfig.deleteOnExit();
311         }
312     }

350     /**
351      * Tests to see if a file is in the system path. <b>Note</b> - the current
352      * implementation only works on non-windows platforms. For purposes of the
353      * AssemblyAnalyzer this is okay as this is only needed on Mac/*nix.
354      *
355      * @param file the executable to look for
356      * @return <code>true</code> if the file exists; otherwise
357      * <code>false</code>
358      */
359     private boolean isInPath(String file) {
360         final ProcessBuilder pb = new ProcessBuilder("which", file);
361         try {
362             final Process proc = pb.start();
363             final int retCode = proc.waitFor();
364             if (retCode == 0) {
365                 return true;
366             }
367         } catch (IOException | InterruptedException ex) {
368             LOGGER.debug("Path search failed for " + file, ex);
369         }
370         return false;
371     }
