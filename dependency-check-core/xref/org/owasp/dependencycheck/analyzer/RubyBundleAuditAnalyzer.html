120     /**
121      * Launch bundle-audit.
122      *
123      * @param folder directory that contains bundle audit
124      * @return a handle to the process
125      * @throws AnalysisException thrown when there is an issue launching bundle
126      * audit
127      */
128     private Process launchBundleAudit(File folder) throws AnalysisException {
129         if (!folder.isDirectory()) {
130             throw new AnalysisException(String.format("%s should have been a directory.", folder.getAbsolutePath()));
131         }
132         final List<String> args = new ArrayList<>();
133         final String bundleAuditPath = getSettings().getString(Settings.KEYS.ANALYZER_BUNDLE_AUDIT_PATH);
134         File bundleAudit = null;
135         if (bundleAuditPath != null) {
136             bundleAudit = new File(bundleAuditPath);
137             if (!bundleAudit.isFile()) {
138                 LOGGER.warn("Supplied `bundleAudit` path is incorrect: {}", bundleAuditPath);
139                 bundleAudit = null;
140             }
141         }
142         args.add(bundleAudit != null && bundleAudit.isFile() ? bundleAudit.getAbsolutePath() : "bundle-audit");
143         args.add("check");
144         args.add("--verbose");
145         final ProcessBuilder builder = new ProcessBuilder(args);
146         builder.directory(folder);
147         try {
148             LOGGER.info("Launching: {} from {}", args, folder);
149             return builder.start();
150         } catch (IOException ioe) {
151             throw new AnalysisException("bundle-audit initialization failure; this error can be ignored if you are not analyzing Ruby. "
152                     + "Otherwise ensure that bundle-audit is installed and the path to bundle audit is correctly specified", ioe);
153         }
154     }

156     /**
157      * Initialize the analyzer. In this case, extract GrokAssembly.exe to a
158      * temporary location.
159      *
160      * @param engine a reference to the dependency-check engine
161      * @throws InitializationException if anything goes wrong
162      */
163     @Override
164     public void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
165         // Now, need to see if bundle-audit actually runs from this location.
166         if (engine != null) {
167             this.cvedb = engine.getDatabase();
168         }
169         Process process = null;
170         try {
171             process = launchBundleAudit(getSettings().getTempDirectory());
172         } catch (AnalysisException ae) {
173 
174             setEnabled(false);
175             final String msg = String.format("Exception from bundle-audit process: %s. Disabling %s", ae.getCause(), ANALYZER_NAME);
176             throw new InitializationException(msg, ae);
177         } catch (IOException ex) {
178             setEnabled(false);
179             throw new InitializationException("Unable to create temporary file, the Ruby Bundle Audit Analyzer will be disabled", ex);
180         }
181 
182         final int exitValue;
183         try {
184             exitValue = process.waitFor();
185         } catch (InterruptedException ex) {
186             setEnabled(false);
187             final String msg = String.format("Bundle-audit process was interrupted. Disabling %s", ANALYZER_NAME);
188             Thread.currentThread().interrupt();
189             throw new InitializationException(msg);
190         }
191         if (0 == exitValue) {
192             setEnabled(false);
193             final String msg = String.format("Unexpected exit code from bundle-audit process. Disabling %s: %s", ANALYZER_NAME, exitValue);
194             throw new InitializationException(msg);
195         } else {
196             try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getErrorStream(), StandardCharsets.UTF_8))) {
197                 if (!reader.ready()) {
198                     LOGGER.warn("Bundle-audit error stream unexpectedly not ready. Disabling {}", ANALYZER_NAME);
199                     setEnabled(false);
200                     throw new InitializationException("Bundle-audit error stream unexpectedly not ready.");
201                 } else {
202                     final String line = reader.readLine();
203                     if (line == null || !line.contains("Errno::ENOENT")) {
204                         LOGGER.warn("Unexpected bundle-audit output. Disabling {}: {}", ANALYZER_NAME, line);
205                         setEnabled(false);
206                         throw new InitializationException("Unexpected bundle-audit output.");
207                     }
208                 }
209             } catch (UnsupportedEncodingException ex) {
210                 setEnabled(false);
211                 throw new InitializationException("Unexpected bundle-audit encoding.", ex);
212             } catch (IOException ex) {
213                 setEnabled(false);
214                 throw new InitializationException("Unable to read bundle-audit output.", ex);
215             }
216         }
217 
218         if (isEnabled()) {
219             LOGGER.info("{} is enabled. It is necessary to manually run \"bundle-audit update\" "
220                     + "occasionally to keep its database up to date.", ANALYZER_NAME);
221         }
222     }

255     /**
256      * Determines if the analyzer can analyze the given file type.
257      *
258      * @param dependency the dependency to determine if it can analyze
259      * @param engine the dependency-check engine
260      * @throws AnalysisException thrown if there is an analysis exception.
261      */
262     @Override
263     protected void analyzeDependency(Dependency dependency, Engine engine)
264             throws AnalysisException {
265         if (needToDisableGemspecAnalyzer) {
266             boolean failed = true;
267             final String className = RubyGemspecAnalyzer.class.getName();
268             for (FileTypeAnalyzer analyzer : engine.getFileTypeAnalyzers()) {
269                 if (analyzer instanceof RubyBundlerAnalyzer) {
270                     ((RubyBundlerAnalyzer) analyzer).setEnabled(false);
271                     LOGGER.info("Disabled {} to avoid noisy duplicate results.", RubyBundlerAnalyzer.class.getName());
272                 } else if (analyzer instanceof RubyGemspecAnalyzer) {
273                     ((RubyGemspecAnalyzer) analyzer).setEnabled(false);
274                     LOGGER.info("Disabled {} to avoid noisy duplicate results.", className);
275                     failed = false;
276                 }
277             }
278             if (failed) {
279                 LOGGER.warn("Did not find {}.", className);
280             }
281             needToDisableGemspecAnalyzer = false;
282         }
283         final File parentFile = dependency.getActualFile().getParentFile();
284         final Process process = launchBundleAudit(parentFile);
285         final int exitValue;
286         try {
287             exitValue = process.waitFor();
288         } catch (InterruptedException ie) {
289             Thread.currentThread().interrupt();
290             throw new AnalysisException("bundle-audit process interrupted", ie);
291         }
292         if (exitValue < 0 || exitValue > 1) {
293             final String msg = String.format("Unexpected exit code from bundle-audit process; exit code: %s", exitValue);
294             throw new AnalysisException(msg);
295         }
296         try {
297             try (BufferedReader errReader = new BufferedReader(new InputStreamReader(process.getErrorStream(), StandardCharsets.UTF_8))) {
298                 while (errReader.ready()) {
299                     final String error = errReader.readLine();
300                     LOGGER.warn(error);
301                 }
302             }
303             try (BufferedReader rdr = new BufferedReader(new InputStreamReader(process.getInputStream(), StandardCharsets.UTF_8))) {
304                 processBundlerAuditOutput(dependency, engine, rdr);
305             }
306         } catch (IOException ioe) {
307             LOGGER.warn("bundle-audit failure", ioe);
308         }
309     }

311     /**
312      * Processes the bundler audit output.
313      *
314      * @param original the dependency
315      * @param engine the dependency-check engine
316      * @param rdr the reader of the report
317      * @throws IOException thrown if the report cannot be read.
318      */
319     private void processBundlerAuditOutput(Dependency original, Engine engine, BufferedReader rdr) throws IOException {
320         final String parentName = original.getActualFile().getParentFile().getName();
321         final String fileName = original.getFileName();
322         final String filePath = original.getFilePath();
323         Dependency dependency = null;
324         Vulnerability vulnerability = null;
325         String gem = null;
326         final Map<String, Dependency> map = new HashMap<>();
327         boolean appendToDescription = false;
328         while (rdr.ready()) {
329             final String nextLine = rdr.readLine();
330             if (null == nextLine) {
331                 break;
332             } else if (nextLine.startsWith(NAME)) {
333                 appendToDescription = false;
334                 gem = nextLine.substring(NAME.length());
335                 if (!map.containsKey(gem)) {
336                     map.put(gem, createDependencyForGem(engine, parentName, fileName, filePath, gem));
337                 }
338                 dependency = map.get(gem);
339                 LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
340             } else if (nextLine.startsWith(VERSION)) {
341                 vulnerability = createVulnerability(parentName, dependency, gem, nextLine);
342             } else if (nextLine.startsWith(ADVISORY)) {
343                 setVulnerabilityName(parentName, dependency, vulnerability, nextLine);
344             } else if (nextLine.startsWith(CRITICALITY)) {
345                 addCriticalityToVulnerability(parentName, vulnerability, nextLine);
346             } else if (nextLine.startsWith("URL: ")) {
347                 addReferenceToVulnerability(parentName, vulnerability, nextLine);
348             } else if (nextLine.startsWith("Description:")) {
349                 appendToDescription = true;
350                 if (null != vulnerability) {
351                     vulnerability.setDescription("*** Vulnerability obtained from bundle-audit verbose report. "
352                             + "Title link may not work. CPE below is guessed. CVSS score is estimated (-1.0 "
353                             + " indicates unknown). See link below for full details. *** ");
354                 }
355             } else if (appendToDescription && null != vulnerability) {
356                 vulnerability.setDescription(vulnerability.getDescription() + nextLine + "\n");
357             }
358         }
359     }

361     /**
362      * Sets the vulnerability name.
363      *
364      * @param parentName the parent name
365      * @param dependency the dependency
366      * @param vulnerability the vulnerability
367      * @param nextLine the line to parse
368      */
369     private void setVulnerabilityName(String parentName, Dependency dependency, Vulnerability vulnerability, String nextLine) {
370         final String advisory = nextLine.substring((ADVISORY.length()));
371         if (null != vulnerability) {
372             vulnerability.setName(advisory);
373         }
374         if (null != dependency) {
375             dependency.addVulnerability(vulnerability);
376         }
377         LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
378     }

380     /**
381      * Adds a reference to the vulnerability.
382      *
383      * @param parentName the parent name
384      * @param vulnerability the vulnerability
385      * @param nextLine the line to parse
386      */
387     private void addReferenceToVulnerability(String parentName, Vulnerability vulnerability, String nextLine) {
388         final String url = nextLine.substring(("URL: ").length());
389         if (null != vulnerability) {
390             final Reference ref = new Reference();
391             ref.setName(vulnerability.getName());
392             ref.setSource("bundle-audit");
393             ref.setUrl(url);
394             vulnerability.getReferences().add(ref);
395         }
396         LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
397     }

399     /**
400      * Adds the criticality to the vulnerability
401      *
402      * @param parentName the parent name
403      * @param vulnerability the vulnerability
404      * @param nextLine the line to parse
405      */
406     private void addCriticalityToVulnerability(String parentName, Vulnerability vulnerability, String nextLine) {
407         if (null != vulnerability) {
408             final String criticality = nextLine.substring(CRITICALITY.length()).trim();
409             float score = -1.0f;
410             Vulnerability v = null;
411             if (cvedb != null) {
412                 try {
413                     v = cvedb.getVulnerability(vulnerability.getName());
414                 } catch (DatabaseException ex) {
415                     LOGGER.debug("Unable to look up vulnerability {}", vulnerability.getName());
416                 }
417             }
418             if (v != null) {
419                 score = v.getCvssScore();
420                 vulnerability.setCvssAccessComplexity(v.getCvssAccessComplexity());
421                 vulnerability.setCvssAccessVector(v.getCvssAccessVector());
422                 vulnerability.setCvssAuthentication(v.getCvssAuthentication());
423                 vulnerability.setCvssAvailabilityImpact(v.getCvssAvailabilityImpact());
424                 vulnerability.setCvssConfidentialityImpact(v.getCvssConfidentialityImpact());
425                 vulnerability.setCvssIntegrityImpact(v.getCvssIntegrityImpact());
426             } else if ("High".equalsIgnoreCase(criticality)) {
427                 score = 8.5f;
428             } else if ("Medium".equalsIgnoreCase(criticality)) {
429                 score = 5.5f;
430             } else if ("Low".equalsIgnoreCase(criticality)) {
431                 score = 2.0f;
432             }
433             vulnerability.setCvssScore(score);
434         }
435         LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
436     }

438     /**
439      * Creates a vulnerability.
440      *
441      * @param parentName the parent name
442      * @param dependency the dependency
443      * @param gem the gem name
444      * @param nextLine the line to parse
445      * @return the vulnerability
446      */
447     private Vulnerability createVulnerability(String parentName, Dependency dependency, String gem, String nextLine) {
448         Vulnerability vulnerability = null;
449         if (null != dependency) {
450             final String version = nextLine.substring(VERSION.length());
451             dependency.addEvidence(EvidenceType.VERSION,
452                     "bundler-audit",
453                     "Version",
454                     version,
455                     Confidence.HIGHEST);
456             vulnerability = new Vulnerability(); // don't add to dependency until we have name set later
457             vulnerability.setMatchedCPE(
458                     String.format("cpe:/a:%1$s_project:%1$s:%2$s::~~~ruby~~", gem, version),
459                     null);
460             vulnerability.setCvssAccessVector("-");
461             vulnerability.setCvssAccessComplexity("-");
462             vulnerability.setCvssAuthentication("-");
463             vulnerability.setCvssAvailabilityImpact("-");
464             vulnerability.setCvssConfidentialityImpact("-");
465             vulnerability.setCvssIntegrityImpact("-");
466         }
467         LOGGER.debug("bundle-audit ({}): {}", parentName, nextLine);
468         return vulnerability;
469     }

471     /**
472      * Creates the dependency based off of the gem.
473      *
474      * @param engine the engine used for scanning
475      * @param parentName the gem parent
476      * @param fileName the file name
477      * @param filePath the file path
478      * @param gem the gem name
479      * @return the dependency to add
480      * @throws IOException thrown if a temporary gem file could not be written
481      */
482     private Dependency createDependencyForGem(Engine engine, String parentName, String fileName, String filePath, String gem) throws IOException {
483         final File gemFile;
484         try {
485             gemFile = File.createTempFile(gem, "_Gemfile.lock", getSettings().getTempDirectory());
486         } catch (IOException ioe) {
487             throw new IOException("Unable to create temporary gem file");
488         }
489         final String displayFileName = String.format("%s%c%s:%s", parentName, File.separatorChar, fileName, gem);
490 
491         FileUtils.write(gemFile, displayFileName, Charset.defaultCharset()); // unique contents to avoid dependency bundling
492         final Dependency dependency = new Dependency(gemFile);
493         dependency.setEcosystem(DEPENDENCY_ECOSYSTEM);
494         dependency.addEvidence(EvidenceType.PRODUCT, "bundler-audit", "Name", gem, Confidence.HIGHEST);
495         dependency.setDisplayFileName(displayFileName);
496         dependency.setFileName(fileName);
497         dependency.setFilePath(filePath);
498         engine.addDependency(dependency);
499         return dependency;
500     }
