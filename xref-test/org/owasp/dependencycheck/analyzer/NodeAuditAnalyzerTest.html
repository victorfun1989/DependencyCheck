33      @Test
34      public void testAnalyzePackage() throws AnalysisException, InitializationException, InvalidSettingException {
35          Assume.assumeThat(getSettings().getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED), is(true));
36          try (Engine engine = new Engine(getSettings())) {
37              NodeAuditAnalyzer analyzer = new NodeAuditAnalyzer();
38              analyzer.setFilesMatched(true);
39              analyzer.initialize(getSettings());
40              analyzer.prepare(engine);
41              final Dependency toScan = new Dependency(BaseTest.getResourceAsFile(this, "nodeaudit/package-lock.json"));
42              analyzer.analyze(toScan, engine);
43              boolean found = false;
44              assertTrue("Mpre then 1 dependency should be identified", 1 < engine.getDependencies().length);
45              for (Dependency result : engine.getDependencies()) {
46                  if ("package-lock.json?uglify-js".equals(result.getFileName())) {
47                      found = true;
48                      assertTrue(result.getEvidence(EvidenceType.VENDOR).toString().contains("uglify-js"));
49                      assertTrue(result.getEvidence(EvidenceType.PRODUCT).toString().contains("uglify-js"));
50                      assertTrue(result.getEvidence(EvidenceType.VERSION).toString().contains("2.4.24"));
51                      assertTrue(result.isVirtual());
52                  }
53              }
54              assertTrue("Uglify was not found", found);
55          }
56      }
