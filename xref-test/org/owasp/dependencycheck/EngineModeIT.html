54              //delete temp files
55              FileUtils.delete(getSettings().getDataDirectory());
56              //Reset system property to original value just to be safe for other tests.
57              System.setProperty(Settings.KEYS.DATA_DIRECTORY, originalDataDir);
58              System.clearProperty(Settings.KEYS.H2_DATA_DIRECTORY);
59          } catch (IOException ex) {
60              throw new RuntimeException(ex);
61          } finally {
62              super.tearDown();
63          }

66      @Test
67      public void testEvidenceCollectionAndEvidenceProcessingModes() throws Exception {
68          Dependency[] dependencies;
69          try (Engine engine = new Engine(Engine.Mode.EVIDENCE_COLLECTION, getSettings())) {
70              engine.openDatabase(); //does nothing in the current mode
71              assertDatabase(false);
72              for (AnalysisPhase phase : Engine.Mode.EVIDENCE_COLLECTION.getPhases()) {
73                  assertThat(engine.getAnalyzers(phase), is(notNullValue()));
74              }
75              for (AnalysisPhase phase : Engine.Mode.EVIDENCE_PROCESSING.getPhases()) {
76                  assertThat(engine.getAnalyzers(phase), is(nullValue()));
77              }
78              File file = BaseTest.getResourceAsFile(this, "struts2-core-2.1.2.jar");
79              engine.scan(file);
80              engine.analyzeDependencies();
81              dependencies = engine.getDependencies();
82              assertThat(dependencies.length, is(1));
83              Dependency dependency = dependencies[0];
84              assertTrue(dependency.getEvidence(EvidenceType.VENDOR).toString().toLowerCase().contains("apache"));
85              assertTrue(dependency.getVendorWeightings().contains("apache"));
86              assertTrue(dependency.getVulnerabilities().isEmpty());
87          }
88  
89          try (Engine engine = new Engine(Engine.Mode.EVIDENCE_PROCESSING, getSettings())) {
90              engine.openDatabase();
91              assertDatabase(true);
92              for (AnalysisPhase phase : Engine.Mode.EVIDENCE_PROCESSING.getPhases()) {
93                  assertThat(engine.getAnalyzers(phase), is(notNullValue()));
94              }
95              for (AnalysisPhase phase : Engine.Mode.EVIDENCE_COLLECTION.getPhases()) {
96                  assertThat(engine.getAnalyzers(phase), is(nullValue()));
97              }
98              engine.addDependency(dependencies[0]);
99              engine.analyzeDependencies();
100             Dependency dependency = dependencies[0];
101             assertFalse(dependency.getVulnerabilities().isEmpty());
102         }
103     }

125     private void assertDatabase(boolean exists) throws Exception {
126         Assume.assumeThat(getSettings().getString(Settings.KEYS.DB_DRIVER_NAME), is("org.h2.Driver"));
127         Path directory = getSettings().getDataDirectory().toPath();
128         assertThat(Files.exists(directory), is(true));
129         assertThat(Files.isDirectory(directory), is(true));
130         Path database = directory.resolve(getSettings().getString(Settings.KEYS.DB_FILE_NAME));
131         assertThat(Files.exists(database), is(exists));
132     }
