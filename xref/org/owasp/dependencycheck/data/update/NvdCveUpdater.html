96      /**
97       * Downloads the latest NVD CVE XML file from the web and imports it into
98       * the current CVE Database. A lock on a file is obtained in an attempt to
99       * prevent more then one thread/JVM from updating the database at the same
100      * time. This method may sleep upto 5 minutes.
101      *
102      * @param engine a reference to the dependency-check engine
103      * @throws UpdateException is thrown if there is an error updating the
104      * database
105      */
106     @Override
107     public synchronized void update(Engine engine) throws UpdateException {
108         this.settings = engine.getSettings();
109         this.cveDb = engine.getDatabase();
110         if (isUpdateConfiguredFalse()) {
111             return;
112         }
113         try {
114             dbProperties = cveDb.getDatabaseProperties();
115             if (checkUpdate()) {
116                 initializeExecutorServices();
117                 final UpdateableNvdCve updateable = getUpdatesNeeded();
118                 if (updateable.isUpdateNeeded()) {
119                     performUpdate(updateable);
120                 }
121                 dbProperties.save(DatabaseProperties.LAST_CHECKED, Long.toString(System.currentTimeMillis()));
122             }
123         } catch (MalformedURLException ex) {
124             throw new UpdateException("NVD CVE properties files contain an invalid URL, unable to update the data to use the most current data.", ex);
125         } catch (DownloadFailedException ex) {
126             LOGGER.warn("Unable to download the NVD CVE data; the results may not include the most recent CPE/CVEs from the NVD.");
127             if (settings.getString(Settings.KEYS.PROXY_SERVER) == null) {
128                 LOGGER.info("If you are behind a proxy you may need to configure dependency-check to use the proxy.");
129             }
130             final String jre = System.getProperty("java.version");
131             if (jre == null || jre.startsWith("1.4") || jre.startsWith("1.5") || jre.startsWith("1.6") || jre.startsWith("1.7")) {
132                 LOGGER.warn("An old JRE is being used ({} {}), and likely does not have the correct root certificates or algorithms "
133                         + "to connect to the NVD - consider upgrading your JRE.", System.getProperty("java.vendor"), jre);
134             }
135             throw new UpdateException("Unable to download the NVD CVE data.", ex);
136         } catch (DatabaseException ex) {
137             throw new UpdateException("Database Exception, unable to update the data to use the most current data.", ex);
138         } finally {
139             shutdownExecutorServices();
140         }
141     }

143     /**
144      * Checks if the system is configured NOT to update.
145      *
146      * @return false if the system is configured to perform an update; otherwise
147      * true
148      */
149     private boolean isUpdateConfiguredFalse() {
150         try {
151             if (!settings.getBoolean(Settings.KEYS.UPDATE_NVDCVE_ENABLED, true)) {
152                 return true;
153             }
154         } catch (InvalidSettingException ex) {
155             LOGGER.trace("invalid setting UPDATE_NVDCVE_ENABLED", ex);
156         }
157         boolean autoUpdate = true;
158         try {
159             autoUpdate = settings.getBoolean(Settings.KEYS.AUTO_UPDATE);
160         } catch (InvalidSettingException ex) {
161             LOGGER.debug("Invalid setting for auto-update; using true.");
162         }
163         return !autoUpdate;
164     }

166     /**
167      * Initialize the executor services for download and processing of the NVD
168      * CVE XML data.
169      */
170     protected void initializeExecutorServices() {
171         processingExecutorService = Executors.newFixedThreadPool(PROCESSING_THREAD_POOL_SIZE);
172         downloadExecutorService = Executors.newFixedThreadPool(DOWNLOAD_THREAD_POOL_SIZE);
173         LOGGER.debug("#download   threads: {}", DOWNLOAD_THREAD_POOL_SIZE);
174         LOGGER.debug("#processing threads: {}", PROCESSING_THREAD_POOL_SIZE);
175     }
176 
177     /**
178      * Shutdown and cleanup of resources used by the executor services.
179      */
180     private void shutdownExecutorServices() {
181         if (processingExecutorService != null) {
182             processingExecutorService.shutdownNow();
183         }
184         if (downloadExecutorService != null) {
185             downloadExecutorService.shutdownNow();
186         }
187     }

189     /**
190      * Checks if the NVD CVE XML files were last checked recently. As an
191      * optimization, we can avoid repetitive checks against the NVD. Setting
192      * CVE_CHECK_VALID_FOR_HOURS determines the duration since last check before
193      * checking again. A database property stores the timestamp of the last
194      * check.
195      *
196      * @return true to proceed with the check, or false to skip
197      * @throws UpdateException thrown when there is an issue checking for
198      * updates
199      */
200     private boolean checkUpdate() throws UpdateException {
201         boolean proceed = true;
202         // If the valid setting has not been specified, then we proceed to check...
203         final int validForHours = settings.getInt(Settings.KEYS.CVE_CHECK_VALID_FOR_HOURS, 0);
204         if (dataExists() && 0 < validForHours) {
205             // ms Valid = valid (hours) x 60 min/hour x 60 sec/min x 1000 ms/sec
206             final long msValid = validForHours * 60L * 60L * 1000L;
207             final long lastChecked = Long.parseLong(dbProperties.getProperty(DatabaseProperties.LAST_CHECKED, "0"));
208             final long now = System.currentTimeMillis();
209             proceed = (now - lastChecked) > msValid;
210             if (!proceed) {
211                 LOGGER.info("Skipping NVD check since last check was within {} hours.", validForHours);
212                 LOGGER.debug("Last NVD was at {}, and now {} is within {} ms.", lastChecked, now, msValid);
213             }
214         }
215         return proceed;
216     }

227     /**
228      * Downloads the latest NVD CVE XML file from the web and imports it into
229      * the current CVE Database.
230      *
231      * @param updateable a collection of NVD CVE data file references that need
232      * to be downloaded and processed to update the database
233      * @throws UpdateException is thrown if there is an error updating the
234      * database
235      */
236     private void performUpdate(UpdateableNvdCve updateable) throws UpdateException {
237         int maxUpdates = 0;
238         for (NvdCveInfo cve : updateable) {
239             if (cve.getNeedsUpdate()) {
240                 maxUpdates += 1;
241             }
242         }
243         if (maxUpdates <= 0) {
244             return;
245         }
246         if (maxUpdates > 3) {
247             LOGGER.info("NVD CVE requires several updates; this could take a couple of minutes.");
248         }
249 
250         final Set<Future<Future<ProcessTask>>> downloadFutures = new HashSet<>(maxUpdates);
251         for (NvdCveInfo cve : updateable) {
252             if (cve.getNeedsUpdate()) {
253                 final DownloadTask call = new DownloadTask(cve, processingExecutorService, cveDb, settings);
254                 downloadFutures.add(downloadExecutorService.submit(call));
255             }
256         }
257 
258         //next, move the future future processTasks to just future processTasks
259         final Set<Future<ProcessTask>> processFutures = new HashSet<>(maxUpdates);
260         for (Future<Future<ProcessTask>> future : downloadFutures) {
261             final Future<ProcessTask> task;
262             try {
263                 task = future.get();
264             } catch (InterruptedException ex) {
265                 LOGGER.debug("Thread was interrupted during download", ex);
266                 Thread.currentThread().interrupt();
267                 throw new UpdateException("The download was interrupted", ex);
268             } catch (ExecutionException ex) {
269                 LOGGER.debug("Thread was interrupted during download execution", ex);
270                 throw new UpdateException("The execution of the download was interrupted", ex);
271             }
272             if (task == null) {
273                 LOGGER.debug("Thread was interrupted during download");
274                 throw new UpdateException("The download was interrupted; unable to complete the update");
275             } else {
276                 processFutures.add(task);
277             }
278         }
279 
280         for (Future<ProcessTask> future : processFutures) {
281             try {
282                 final ProcessTask task = future.get();
283                 if (task.getException() != null) {
284                     throw task.getException();
285                 }
286             } catch (InterruptedException ex) {
287                 LOGGER.debug("Thread was interrupted during processing", ex);
288                 Thread.currentThread().interrupt();
289                 throw new UpdateException(ex);
290             } catch (ExecutionException ex) {
291                 LOGGER.debug("Execution Exception during process", ex);
292                 throw new UpdateException(ex);
293             }
294         }
295 
296         //always true because <=0 exits early above
297         //if (maxUpdates >= 1) {
298         //ensure the modified file date gets written (we may not have actually updated it)
299         dbProperties.save(updateable.get(MODIFIED));
300         LOGGER.info("Begin database maintenance.");
301         cveDb.cleanupDatabase();
302         LOGGER.info("End database maintenance.");
303         //}
304     }

306     /**
307      * Determines if the index needs to be updated. This is done by fetching the
308      * NVD CVE meta data and checking the last update date. If the data needs to
309      * be refreshed this method will return the NvdCveUrl for the files that
310      * need to be updated.
311      *
312      * @return the collection of files that need to be updated
313      * @throws MalformedURLException is thrown if the URL for the NVD CVE Meta
314      * data is incorrect
315      * @throws DownloadFailedException is thrown if there is an error.
316      * downloading the NVD CVE download data file
317      * @throws UpdateException Is thrown if there is an issue with the last
318      * updated properties file
319      */
320     protected final UpdateableNvdCve getUpdatesNeeded() throws MalformedURLException, DownloadFailedException, UpdateException {
321         LOGGER.info("starting getUpdatesNeeded() ...");
322         final UpdateableNvdCve updates;
323         try {
324             updates = retrieveCurrentTimestampsFromWeb();
325         } catch (InvalidDataException ex) {
326             final String msg = "Unable to retrieve valid timestamp from nvd cve downloads page";
327             LOGGER.debug(msg, ex);
328             throw new DownloadFailedException(msg, ex);
329         } catch (InvalidSettingException ex) {
330             LOGGER.debug("Invalid setting found when retrieving timestamps", ex);
331             throw new DownloadFailedException("Invalid settings", ex);
332         }
333 
334         if (updates == null) {
335             throw new DownloadFailedException("Unable to retrieve the timestamps of the currently published NVD CVE data");
336         }
337         if (dbProperties != null && !dbProperties.isEmpty()) {
338             try {
339                 final int startYear = settings.getInt(Settings.KEYS.CVE_START_YEAR, 2002);
340                 final int endYear = Calendar.getInstance().get(Calendar.YEAR);
341                 boolean needsFullUpdate = false;
342                 for (int y = startYear; y <= endYear; y++) {
343                     final long val = Long.parseLong(dbProperties.getProperty(DatabaseProperties.LAST_UPDATED_BASE + y, "0"));
344                     if (val == 0) {
345                         needsFullUpdate = true;
346                     }
347                 }
348 
349                 final long lastUpdated = Long.parseLong(dbProperties.getProperty(DatabaseProperties.LAST_UPDATED, "0"));
350                 final long now = System.currentTimeMillis();
351                 final int days = settings.getInt(Settings.KEYS.CVE_MODIFIED_VALID_FOR_DAYS, 7);
352                 if (!needsFullUpdate && lastUpdated == updates.getTimeStamp(MODIFIED)) {
353                     updates.clear(); //we don't need to update anything.
354                 } else if (!needsFullUpdate && DateUtil.withinDateRange(lastUpdated, now, days)) {
355                     for (NvdCveInfo entry : updates) {
356                         if (MODIFIED.equals(entry.getId())) {
357                             entry.setNeedsUpdate(true);
358                         } else {
359                             entry.setNeedsUpdate(false);
360                         }
361                     }
362                 } else { //we figure out which of the several XML files need to be downloaded.
363                     for (NvdCveInfo entry : updates) {
364                         if (MODIFIED.equals(entry.getId())) {
365                             entry.setNeedsUpdate(true);
366                         } else {
367                             long currentTimestamp = 0;
368                             try {
369                                 currentTimestamp = Long.parseLong(dbProperties.getProperty(DatabaseProperties.LAST_UPDATED_BASE
370                                         + entry.getId(), "0"));
371                             } catch (NumberFormatException ex) {
372                                 LOGGER.debug("Error parsing '{}' '{}' from nvdcve.lastupdated",
373                                         DatabaseProperties.LAST_UPDATED_BASE, entry.getId(), ex);
374                             }
375                             if (currentTimestamp == entry.getTimestamp()) {
376                                 entry.setNeedsUpdate(false);
377                             }
378                         }
379                     }
380                 }
381             } catch (NumberFormatException ex) {
382                 LOGGER.warn("An invalid schema version or timestamp exists in the data.properties file.");
383                 LOGGER.debug("", ex);
384             }
385         }
386         return updates;
387     }

389     /**
390      * Retrieves the timestamps from the NVD CVE meta data file.
391      *
392      * @return the timestamp from the currently published nvdcve downloads page
393      * @throws MalformedURLException thrown if the URL for the NVD CCE Meta data
394      * is incorrect.
395      * @throws DownloadFailedException thrown if there is an error downloading
396      * the nvd cve meta data file
397      * @throws InvalidDataException thrown if there is an exception parsing the
398      * timestamps
399      * @throws InvalidSettingException thrown if the settings are invalid
400      */
401     private UpdateableNvdCve retrieveCurrentTimestampsFromWeb()
402             throws MalformedURLException, DownloadFailedException, InvalidDataException, InvalidSettingException {
403 
404         final int start = settings.getInt(Settings.KEYS.CVE_START_YEAR);
405         final int end = Calendar.getInstance().get(Calendar.YEAR);
406 
407         final Map<String, Long> lastModifiedDates = retrieveLastModifiedDates(start, end);
408 
409         final UpdateableNvdCve updates = new UpdateableNvdCve();
410 
411         final String baseUrl20 = settings.getString(Settings.KEYS.CVE_SCHEMA_2_0);
412         final String baseUrl12 = settings.getString(Settings.KEYS.CVE_SCHEMA_1_2);
413         for (int i = start; i <= end; i++) {
414             final String url = String.format(baseUrl20, i);
415             updates.add(Integer.toString(i), url, String.format(baseUrl12, i),
416                     lastModifiedDates.get(url), true);
417         }
418 
419         final String url = settings.getString(Settings.KEYS.CVE_MODIFIED_20_URL);
420         updates.add(MODIFIED, url, settings.getString(Settings.KEYS.CVE_MODIFIED_12_URL),
421                 lastModifiedDates.get(url), false);
422         return updates;
423     }

425     /**
426      * Retrieves the timestamps from the NVD CVE meta data file.
427      *
428      * @param startYear the first year whose item to check for the timestamp
429      * @param endYear the last year whose item to check for the timestamp
430      * @return the timestamps from the currently published NVD CVE downloads
431      * page
432      * @throws MalformedURLException thrown if the URL for the NVD CCE Meta data
433      * is incorrect.
434      * @throws DownloadFailedException thrown if there is an error downloading
435      * the NVD CVE meta data file
436      */
437     private Map<String, Long> retrieveLastModifiedDates(int startYear, int endYear)
438             throws MalformedURLException, DownloadFailedException {
439 
440         final Set<String> urls = new HashSet<>();
441         final String baseUrl20 = settings.getString(Settings.KEYS.CVE_SCHEMA_2_0);
442         for (int i = startYear; i <= endYear; i++) {
443             final String url = String.format(baseUrl20, i);
444             urls.add(url);
445         }
446         urls.add(settings.getString(Settings.KEYS.CVE_MODIFIED_20_URL));
447 
448         final Map<String, Future<Long>> timestampFutures = new HashMap<>();
449         for (String url : urls) {
450             final TimestampRetriever timestampRetriever = new TimestampRetriever(url, settings);
451             final Future<Long> future = downloadExecutorService.submit(timestampRetriever);
452             timestampFutures.put(url, future);
453         }
454 
455         final Map<String, Long> lastModifiedDates = new HashMap<>();
456         for (String url : urls) {
457             final Future<Long> timestampFuture = timestampFutures.get(url);
458             final long timestamp;
459             try {
460                 timestamp = timestampFuture.get(60, TimeUnit.SECONDS);
461             } catch (InterruptedException e) {
462                 Thread.currentThread().interrupt();
463                 throw new DownloadFailedException(e);
464             } catch (ExecutionException | TimeoutException e) {
465                 throw new DownloadFailedException(e);
466             }
467             lastModifiedDates.put(url, timestamp);
468         }
469 
470         return lastModifiedDates;
471     }

482     /**
483      * Retrieves the last modified timestamp from a NVD CVE meta data file.
484      */
485     private static class TimestampRetriever implements Callable<Long> {
486 
487         /**
488          * A reference to the global settings object.
489          */
490         private final Settings settings;
491         /**
492          * The URL to obtain the timestamp from.
493          */
494         private final String url;
495 
496         /**
497          * Instantiates a new timestamp retriever object.
498          *
499          * @param url the URL to hit
500          * @param settings the global settings
501          */
502         TimestampRetriever(String url, Settings settings) {
503             this.url = url;
504             this.settings = settings;
505         }
506 
507         @Override
508         public Long call() throws Exception {
509             LOGGER.debug("Checking for updates from: {}", url);
510             try {
511                 final Downloader downloader = new Downloader(settings);
512                 return downloader.getLastModified(new URL(url));
513             } finally {
514                 settings.cleanup(false);
515             }
516         }
517     }
