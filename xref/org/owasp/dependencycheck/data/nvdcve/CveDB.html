205     private String determineDatabaseProductName(Connection conn) {

207             final String databaseProductName = conn.getMetaData().getDatabaseProductName().toLowerCase();
208             LOGGER.debug("Database product: {}", databaseProductName);

216     /**
217      * Opens the database connection. If the database does not exist, it will
218      * create a new one.
219      *
220      * @throws DatabaseException thrown if there is an error opening the
221      * database connection
222      */
223     private synchronized void open() throws DatabaseException {
224         try {
225             if (!isOpen()) {
226                 connection = connectionFactory.getConnection();
227                 final String databaseProductName = determineDatabaseProductName(this.connection);
228                 statementBundle = databaseProductName != null
229                         ? ResourceBundle.getBundle("data/dbStatements", new Locale(databaseProductName))
230                         : ResourceBundle.getBundle("data/dbStatements");
231                 prepareStatements();
232                 databaseProperties = new DatabaseProperties(this);
233             }
234         } catch (DatabaseException e) {
235             releaseResources();
236             throw e;
237         }
238     }
239 
240     /**
241      * Closes the database connection. Close should be called on this object
242      * when it is done being used.
243      */
244     @Override
245     public synchronized void close() {
246         if (isOpen()) {
247             clearCache();
248             closeStatements();
249             try {
250                 connection.close();
251             } catch (SQLException ex) {
252                 LOGGER.error("There was an error attempting to close the CveDB, see the log for more details.");
253                 LOGGER.debug("", ex);
254             } catch (Throwable ex) {
255                 LOGGER.error("There was an exception attempting to close the CveDB, see the log for more details.");
256                 LOGGER.debug("", ex);
257             }
258             releaseResources();
259             connectionFactory.cleanup();
260         }
261     }

282     /**
283      * Prepares all statements to be used.
284      *
285      * @throws DatabaseException thrown if there is an error preparing the
286      * statements
287      */
288     private void prepareStatements() throws DatabaseException {
289         for (PreparedStatementCveDb key : values()) {
290             PreparedStatement preparedStatement = null;
291             try {
292                 final String statementString = statementBundle.getString(key.name());
293                 if (key == INSERT_VULNERABILITY || key == INSERT_CPE) {
294                     preparedStatement = connection.prepareStatement(statementString, new String[]{"id"});
295                 } else {
296                     preparedStatement = connection.prepareStatement(statementString);
297                 }
298             } catch (SQLException ex) {
299                 throw new DatabaseException(ex);
300             } catch (MissingResourceException ex) {
301                 if (!ex.getMessage().contains("key MERGE_PROPERTY")) {
302                     throw new DatabaseException(ex);
303                 }
304             }
305             if (preparedStatement != null) {
306                 preparedStatements.put(key, preparedStatement);
307             }
308         }
309     }

311     /**
312      * Closes all prepared statements.
313      */
314     private synchronized void closeStatements() {
315         for (PreparedStatement preparedStatement : preparedStatements.values()) {
316             DBUtils.closeStatement(preparedStatement);
317         }
318     }

321      * Returns the specified prepared statement.

324      * return
325      * @return the prepared statement
326      * @throws SQLException thrown if a SQL Exception occurs
327      */
328     private synchronized PreparedStatement getPreparedStatement(PreparedStatementCveDb key) throws SQLException {
329         if (!preparedStatements.containsKey(key)) {
330             return null;
331         }
332         final PreparedStatement preparedStatement = preparedStatements.get(key);
333         preparedStatement.clearParameters();
334         return preparedStatement;
335     }

382     /**
383      * Searches the CPE entries in the database and retrieves all entries for a
384      * given vendor and product combination. The returned list will include all
385      * versions of the product that are registered in the NVD CVE data.
386      *
387      * @param vendor the identified vendor name of the dependency being analyzed
388      * @param product the identified name of the product of the dependency being
389      * analyzed
390      * @return a set of vulnerable software
391      */
392     public synchronized Set<VulnerableSoftware> getCPEs(String vendor, String product) {
393         final Set<VulnerableSoftware> cpe = new HashSet<>();
394         ResultSet rs = null;
395         try {
396             final PreparedStatement ps = getPreparedStatement(SELECT_CPE_ENTRIES);
397             ps.setString(1, vendor);
398             ps.setString(2, product);
399             rs = ps.executeQuery();
400 
401             while (rs.next()) {
402                 final VulnerableSoftware vs = new VulnerableSoftware();
403                 vs.setCpe(rs.getString(1));
404                 cpe.add(vs);
405             }
406         } catch (SQLException ex) {
407             LOGGER.error("An unexpected SQL Exception occurred; please see the verbose log for more details.");
408             LOGGER.debug("", ex);
409         } finally {
410             DBUtils.closeResultSet(rs);
411         }
412         return cpe;
413     }

463     /**
464      * Saves a property to the database.
465      *
466      * @param key the property key
467      * @param value the property value
468      */
469     public synchronized void saveProperty(String key, String value) {
470         clearCache();
471         try {
472             final PreparedStatement mergeProperty = getPreparedStatement(MERGE_PROPERTY);
473             if (mergeProperty != null) {
474                 mergeProperty.setString(1, key);
475                 mergeProperty.setString(2, value);
476                 mergeProperty.execute();
477             } else {
478                 // No Merge statement, so doing an Update/Insert...
479                 final PreparedStatement updateProperty = getPreparedStatement(UPDATE_PROPERTY);
480                 updateProperty.setString(1, value);
481                 updateProperty.setString(2, key);
482                 if (updateProperty.executeUpdate() == 0) {
483                     final PreparedStatement insertProperty = getPreparedStatement(INSERT_PROPERTY);
484                     insertProperty.setString(1, key);
485                     insertProperty.setString(2, value);
486                     insertProperty.executeUpdate();
487                 }
488             }
489         } catch (SQLException ex) {
490             LOGGER.warn("Unable to save property '{}' with a value of '{}' to the database", key, value);
491             LOGGER.debug("", ex);
492         }
493     }

508     /**
509      * Retrieves the vulnerabilities associated with the specified CPE.
510      *
511      * @param cpeStr the CPE name
512      * @return a list of Vulnerabilities
513      * @throws DatabaseException thrown if there is an exception retrieving data
514      */
515     public synchronized List<Vulnerability> getVulnerabilities(String cpeStr) throws DatabaseException {
516         final List<Vulnerability> cachedVulnerabilities = vulnerabilitiesForCpeCache.get(cpeStr);
517         if (cachedVulnerabilities != null) {
518             LOGGER.debug("Cache hit for {}", cpeStr);
519             return cachedVulnerabilities;
520         } else {
521             LOGGER.debug("Cache miss for {}", cpeStr);
522         }
523         final VulnerableSoftware cpe = new VulnerableSoftware();
524         try {
525             cpe.parseName(cpeStr);
526         } catch (UnsupportedEncodingException ex) {
527             LOGGER.trace("", ex);
528         }
529         final DependencyVersion detectedVersion = parseDependencyVersion(cpe);
530         final List<Vulnerability> vulnerabilities = new ArrayList<>();
531 
532         ResultSet rs = null;
533         try {
534             final PreparedStatement ps = getPreparedStatement(SELECT_CVE_FROM_SOFTWARE);
535             ps.setString(1, cpe.getVendor());
536             ps.setString(2, cpe.getProduct());
537             rs = ps.executeQuery();
538             String currentCVE = "";
539 
540             final Map<String, Boolean> vulnSoftware = new HashMap<>();
541             while (rs.next()) {
542                 final String cveId = rs.getString(1);
543                 if (!currentCVE.equals(cveId)) { //check for match and add
544                     final Entry<String, Boolean> matchedCPE = getMatchingSoftware(vulnSoftware, cpe.getVendor(), cpe.getProduct(), detectedVersion);
545                     if (matchedCPE != null) {
546                         final Vulnerability v = getVulnerability(currentCVE);
547                         if (v != null) {
548                             v.setMatchedCPE(matchedCPE.getKey(), matchedCPE.getValue() ? "Y" : null);
549                             vulnerabilities.add(v);
550                         }
551                     }
552                     vulnSoftware.clear();
553                     currentCVE = cveId;
554                 }
555 
556                 final String cpeId = rs.getString(2);
557                 final String previous = rs.getString(3);
558                 final Boolean p = previous != null && !previous.isEmpty();
559                 vulnSoftware.put(cpeId, p);
560             }
561             //remember to process the last set of CVE/CPE entries
562             final Entry<String, Boolean> matchedCPE = getMatchingSoftware(vulnSoftware, cpe.getVendor(), cpe.getProduct(), detectedVersion);
563             if (matchedCPE != null) {
564                 final Vulnerability v = getVulnerability(currentCVE);
565                 if (v != null) {
566                     v.setMatchedCPE(matchedCPE.getKey(), matchedCPE.getValue() ? "Y" : null);
567                     vulnerabilities.add(v);
568                 }
569             }
570         } catch (SQLException ex) {
571             throw new DatabaseException("Exception retrieving vulnerability for " + cpeStr, ex);
572         } finally {
573             DBUtils.closeResultSet(rs);
574         }
575         vulnerabilitiesForCpeCache.put(cpeStr, vulnerabilities);
576         return vulnerabilities;
577     }

579     /**
580      * Gets a vulnerability for the provided CVE.
581      *
582      * @param cve the CVE to lookup
583      * @return a vulnerability object
584      * @throws DatabaseException if an exception occurs
585      */
586     public synchronized Vulnerability getVulnerability(String cve) throws DatabaseException {
587         ResultSet rsV = null;
588         ResultSet rsR = null;
589         ResultSet rsS = null;
590         Vulnerability vuln = null;
591 
592         try {
593             final PreparedStatement psV = getPreparedStatement(SELECT_VULNERABILITY);
594             psV.setString(1, cve);
595             rsV = psV.executeQuery();
596             if (rsV.next()) {
597                 vuln = new Vulnerability();
598                 vuln.setName(cve);
599                 vuln.setDescription(rsV.getString(2));
600                 String cwe = rsV.getString(3);
601                 if (cwe != null) {
602                     final String name = CweDB.getCweName(cwe);
603                     if (name != null) {
604                         cwe += ' ' + name;
605                     }
606                 }
607                 final int cveId = rsV.getInt(1);
608                 vuln.setCwe(cwe);
609                 vuln.setCvssScore(rsV.getFloat(4));
610                 vuln.setCvssAccessVector(rsV.getString(5));
611                 vuln.setCvssAccessComplexity(rsV.getString(6));
612                 vuln.setCvssAuthentication(rsV.getString(7));
613                 vuln.setCvssConfidentialityImpact(rsV.getString(8));
614                 vuln.setCvssIntegrityImpact(rsV.getString(9));
615                 vuln.setCvssAvailabilityImpact(rsV.getString(10));
616 
617                 final PreparedStatement psR = getPreparedStatement(SELECT_REFERENCES);
618                 psR.setInt(1, cveId);
619                 rsR = psR.executeQuery();
620                 while (rsR.next()) {
621                     vuln.addReference(rsR.getString(1), rsR.getString(2), rsR.getString(3));
622                 }
623 
624                 final PreparedStatement psS = getPreparedStatement(SELECT_SOFTWARE);
625                 psS.setInt(1, cveId);
626                 rsS = psS.executeQuery();
627                 while (rsS.next()) {
628                     final String cpe = rsS.getString(1);
629                     final String prevVersion = rsS.getString(2);
630                     if (prevVersion == null) {
631                         vuln.addVulnerableSoftware(cpe);
632                     } else {
633                         vuln.addVulnerableSoftware(cpe, prevVersion);
634                     }
635                 }
636             }
637         } catch (SQLException ex) {
638             throw new DatabaseException("Error retrieving " + cve, ex);
639         } finally {
640             DBUtils.closeResultSet(rsV);
641             DBUtils.closeResultSet(rsR);
642             DBUtils.closeResultSet(rsS);
643         }
644         return vuln;
645     }

647     /**
648      * Updates the vulnerability within the database. If the vulnerability does
649      * not exist it will be added.
650      *
651      * @param vuln the vulnerability to add to the database
652      * @throws DatabaseException is thrown if the database
653      */
654     public synchronized void updateVulnerability(Vulnerability vuln) throws DatabaseException {
655         clearCache();
656         ResultSet rs = null;
657         try {
658             int vulnerabilityId = 0;
659             final PreparedStatement selectVulnerabilityId = getPreparedStatement(SELECT_VULNERABILITY_ID);
660             selectVulnerabilityId.setString(1, vuln.getName());
661             rs = selectVulnerabilityId.executeQuery();
662             if (rs.next()) {
663                 vulnerabilityId = rs.getInt(1);
664                 // first delete any existing vulnerability info. We don't know what was updated. yes, slower but atm easier.
665                 final PreparedStatement deleteReference = getPreparedStatement(DELETE_REFERENCE);
666                 deleteReference.setInt(1, vulnerabilityId);
667                 deleteReference.execute();
668 
669                 final PreparedStatement deleteSoftware = getPreparedStatement(DELETE_SOFTWARE);
670                 deleteSoftware.setInt(1, vulnerabilityId);
671                 deleteSoftware.execute();
672             }
673 
674             DBUtils.closeResultSet(rs);
675 
676             if (vulnerabilityId != 0) {
677                 if (vuln.getDescription().contains("** REJECT **")) {
678                     final PreparedStatement deleteVulnerability = getPreparedStatement(DELETE_VULNERABILITY);
679                     deleteVulnerability.setInt(1, vulnerabilityId);
680                     deleteVulnerability.executeUpdate();
681                 } else {
682                     final PreparedStatement updateVulnerability = getPreparedStatement(UPDATE_VULNERABILITY);
683                     updateVulnerability.setString(1, vuln.getDescription());
684                     updateVulnerability.setString(2, vuln.getCwe());
685                     updateVulnerability.setFloat(3, vuln.getCvssScore());
686                     updateVulnerability.setString(4, vuln.getCvssAccessVector());
687                     updateVulnerability.setString(5, vuln.getCvssAccessComplexity());
688                     updateVulnerability.setString(6, vuln.getCvssAuthentication());
689                     updateVulnerability.setString(7, vuln.getCvssConfidentialityImpact());
690                     updateVulnerability.setString(8, vuln.getCvssIntegrityImpact());
691                     updateVulnerability.setString(9, vuln.getCvssAvailabilityImpact());
692                     updateVulnerability.setInt(10, vulnerabilityId);
693                     updateVulnerability.executeUpdate();
694                 }
695             } else {
696                 final PreparedStatement insertVulnerability = getPreparedStatement(INSERT_VULNERABILITY);
697                 insertVulnerability.setString(1, vuln.getName());
698                 insertVulnerability.setString(2, vuln.getDescription());
699                 insertVulnerability.setString(3, vuln.getCwe());
700                 insertVulnerability.setFloat(4, vuln.getCvssScore());
701                 insertVulnerability.setString(5, vuln.getCvssAccessVector());
702                 insertVulnerability.setString(6, vuln.getCvssAccessComplexity());
703                 insertVulnerability.setString(7, vuln.getCvssAuthentication());
704                 insertVulnerability.setString(8, vuln.getCvssConfidentialityImpact());
705                 insertVulnerability.setString(9, vuln.getCvssIntegrityImpact());
706                 insertVulnerability.setString(10, vuln.getCvssAvailabilityImpact());
707                 insertVulnerability.execute();
708                 try {
709                     rs = insertVulnerability.getGeneratedKeys();
710                     rs.next();
711                     vulnerabilityId = rs.getInt(1);
712                 } catch (SQLException ex) {
713                     final String msg = String.format("Unable to retrieve id for new vulnerability for '%s'", vuln.getName());
714                     throw new DatabaseException(msg, ex);
715                 } finally {
716                     DBUtils.closeResultSet(rs);
717                 }
718             }
719 
720             PreparedStatement insertReference = getPreparedStatement(INSERT_REFERENCE);
721             int countReferences = 0;
722             for (Reference r : vuln.getReferences()) {
723                 insertReference.setInt(1, vulnerabilityId);
724                 insertReference.setString(2, r.getName());
725                 insertReference.setString(3, r.getUrl());
726                 insertReference.setString(4, r.getSource());
727                 if (isBatchInsertEnabled()) {
728                     insertReference.addBatch();
729                     countReferences++;
730                     if (countReferences % getBatchSize() == 0) {
731                         insertReference.executeBatch();
732                         insertReference = getPreparedStatement(INSERT_REFERENCE);
733                         LOGGER.trace(getLogForBatchInserts(countReferences, "Completed %s batch inserts to references table: %s"));
734                         countReferences = 0;
735                     } else if (countReferences == vuln.getReferences().size()) {
736                         if (LOGGER.isTraceEnabled()) {
737                             LOGGER.trace(getLogForBatchInserts(countReferences, "Completed %s batch inserts to reference table: %s"));
738                         }
739                         insertReference.executeBatch();
740                         countReferences = 0;
741                     }
742                 } else {
743                     insertReference.execute();
744                 }
745             }
746 
747             PreparedStatement insertSoftware = getPreparedStatement(INSERT_SOFTWARE);
748             int countSoftware = 0;
749             for (VulnerableSoftware vulnerableSoftware : vuln.getVulnerableSoftware()) {
750                 int cpeProductId = 0;
751                 final PreparedStatement selectCpeId = getPreparedStatement(SELECT_CPE_ID);
752                 selectCpeId.setString(1, vulnerableSoftware.getName());
753                 try {
754                     rs = selectCpeId.executeQuery();
755                     if (rs.next()) {
756                         cpeProductId = rs.getInt(1);
757                     }
758                 } catch (SQLException ex) {
759                     throw new DatabaseException("Unable to get primary key for new cpe: " + vulnerableSoftware.getName(), ex);
760                 } finally {
761                     DBUtils.closeResultSet(rs);
762                 }
763 
764                 if (cpeProductId == 0) {
765                     final PreparedStatement insertCpe = getPreparedStatement(INSERT_CPE);
766                     insertCpe.setString(1, vulnerableSoftware.getName());
767                     insertCpe.setString(2, vulnerableSoftware.getVendor());
768                     insertCpe.setString(3, vulnerableSoftware.getProduct());
769                     insertCpe.executeUpdate();
770                     cpeProductId = DBUtils.getGeneratedKey(insertCpe);
771                 }
772                 if (cpeProductId == 0) {
773                     throw new DatabaseException("Unable to retrieve cpeProductId - no data returned");
774                 }
775 
776                 insertSoftware.setInt(1, vulnerabilityId);
777                 insertSoftware.setInt(2, cpeProductId);
778 
779                 if (vulnerableSoftware.getPreviousVersion() == null) {
780                     insertSoftware.setNull(3, java.sql.Types.VARCHAR);
781                 } else {
782                     insertSoftware.setString(3, vulnerableSoftware.getPreviousVersion());
783                 }
784                 if (isBatchInsertEnabled()) {
785                     insertSoftware.addBatch();
786                     countSoftware++;
787                     if (countSoftware % getBatchSize() == 0) {
788                         executeBatch(vuln, vulnerableSoftware, insertSoftware);
789                         insertSoftware = getPreparedStatement(INSERT_SOFTWARE);
790                         LOGGER.trace(getLogForBatchInserts(countSoftware, "Completed %s batch inserts software table: %s"));
791                         countSoftware = 0;
792                     } else if (countSoftware == vuln.getVulnerableSoftware().size()) {
793                         if (LOGGER.isTraceEnabled()) {
794                             LOGGER.trace(getLogForBatchInserts(countSoftware, "Completed %s batch inserts software table: %s"));
795                             countReferences = 0;
796                         }
797                         executeBatch(vuln, vulnerableSoftware, insertSoftware);
798                     }
799                 } else {
800                     try {
801                         insertSoftware.execute();
802                     } catch (SQLException ex) {
803                         if (ex.getMessage().contains("Duplicate entry")) {
804                             final String msg = String.format("Duplicate software key identified in '%s:%s'", vuln.getName(), vuln.getName());
805                             LOGGER.info(msg, ex);
806                         } else {
807                             throw ex;
808                         }
809                     }
810                 }
811             }
812         } catch (SQLException ex) {
813             final String msg = String.format("Error updating '%s'", vuln.getName());
814             LOGGER.debug(msg, ex);
815             throw new DatabaseException(msg, ex);
816         } finally {
817             DBUtils.closeResultSet(rs);
818         }
819     }

938     /**
939      * Determines if the given identifiedVersion is affected by the given cpeId
940      * and previous version flag. A non-null, non-empty string passed to the
941      * previous version argument indicates that all previous versions are
942      * affected.
943      *
944      * @param vendor the vendor of the dependency being analyzed
945      * @param product the product name of the dependency being analyzed
946      * @param vulnerableSoftware a map of the vulnerable software with a boolean
947      * indicating if all previous versions are affected
948      * @param identifiedVersion the identified version of the dependency being
949      * analyzed
950      * @return true if the identified version is affected, otherwise false
951      */
952     protected Entry<String, Boolean> getMatchingSoftware(Map<String, Boolean> vulnerableSoftware, String vendor, String product,
953             DependencyVersion identifiedVersion) {
954 
955         final boolean isVersionTwoADifferentProduct = "apache".equals(vendor) && "struts".equals(product);
956 
957         final Set<String> majorVersionsAffectingAllPrevious = new HashSet<>();
958         final boolean matchesAnyPrevious = identifiedVersion == null || "-".equals(identifiedVersion.toString());
959         String majorVersionMatch = null;
960         for (Entry<String, Boolean> entry : vulnerableSoftware.entrySet()) {
961             final DependencyVersion v = parseDependencyVersion(entry.getKey());
962             if (v == null || "-".equals(v.toString())) { //all versions
963                 return entry;
964             }
965             if (entry.getValue()) {
966                 if (matchesAnyPrevious) {
967                     return entry;
968                 }
969                 if (identifiedVersion != null && identifiedVersion.getVersionParts().get(0).equals(v.getVersionParts().get(0))) {
970                     majorVersionMatch = v.getVersionParts().get(0);
971                 }
972                 majorVersionsAffectingAllPrevious.add(v.getVersionParts().get(0));
973             }
974         }
975         if (matchesAnyPrevious) {
976             return null;
977         }
978 
979         final boolean canSkipVersions = majorVersionMatch != null && majorVersionsAffectingAllPrevious.size() > 1;
980         //yes, we are iterating over this twice. The first time we are skipping versions those that affect all versions
981         //then later we process those that affect all versions. This could be done with sorting...
982         for (Entry<String, Boolean> entry : vulnerableSoftware.entrySet()) {
983             if (!entry.getValue()) {
984                 final DependencyVersion v = parseDependencyVersion(entry.getKey());
985                 //this can't dereference a null 'majorVersionMatch' as canSkipVersions accounts for this.
986                 if (canSkipVersions && majorVersionMatch != null && !majorVersionMatch.equals(v.getVersionParts().get(0))) {
987                     continue;
988                 }
989                 //this can't dereference a null 'identifiedVersion' because if it was null we would have exited
990                 //in the above loop or just after loop (if matchesAnyPrevious return null).
991                 if (identifiedVersion != null && identifiedVersion.equals(v)) {
992                     return entry;
993                 }
994             }
995         }
996         for (Entry<String, Boolean> entry : vulnerableSoftware.entrySet()) {
997             if (entry.getValue()) {
998                 final DependencyVersion v = parseDependencyVersion(entry.getKey());
999                 //this can't dereference a null 'majorVersionMatch' as canSkipVersions accounts for this.
1000                 if (canSkipVersions && majorVersionMatch != null && !majorVersionMatch.equals(v.getVersionParts().get(0))) {
1001                     continue;
1002                 }
1003                 //this can't dereference a null 'identifiedVersion' because if it was null we would have exited
1004                 //in the above loop or just after loop (if matchesAnyPrevious return null).
1005                 if (entry.getValue() && identifiedVersion != null && identifiedVersion.compareTo(v) <= 0
1006                         && !(isVersionTwoADifferentProduct && !identifiedVersion.getVersionParts().get(0).equals(v.getVersionParts().get(0)))) {
1007                     return entry;
1008                 }
1009             }
1010         }
1011         return null;
1012     }

1014     /**
1015      * Parses the version (including revision) from a CPE identifier. If no
1016      * version is identified then a '-' is returned.
1017      *
1018      * @param cpeStr a cpe identifier
1019      * @return a dependency version
1020      */
1021     private DependencyVersion parseDependencyVersion(String cpeStr) {
1022         final VulnerableSoftware cpe = new VulnerableSoftware();
1023         try {
1024             cpe.parseName(cpeStr);
1025         } catch (UnsupportedEncodingException ex) {
1026             //never going to happen.
1027             LOGGER.trace("", ex);
1028         }
1029         return parseDependencyVersion(cpe);
1030     }

1032     /**
1033      * Takes a CPE and parses out the version number. If no version is
1034      * identified then a '-' is returned.
1035      *
1036      * @param cpe a cpe object
1037      * @return a dependency version
1038      */
1039     private DependencyVersion parseDependencyVersion(VulnerableSoftware cpe) {
1040         final DependencyVersion cpeVersion;
1041         if (cpe.getVersion() != null && !cpe.getVersion().isEmpty()) {
1042             final String versionText;
1043             if (cpe.getUpdate() != null && !cpe.getUpdate().isEmpty()) {
1044                 versionText = String.format("%s.%s", cpe.getVersion(), cpe.getUpdate());
1045             } else {
1046                 versionText = cpe.getVersion();
1047             }
1048             cpeVersion = DependencyVersionUtil.parseVersion(versionText, true);
1049         } else {
1050             cpeVersion = new DependencyVersion("-");
1051         }
1052         return cpeVersion;
1053     }
