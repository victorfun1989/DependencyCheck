900     /**
901      * Resolves the projects artifacts using Aether and scans the resulting
902      * dependencies.
903      *
904      * @param engine the core dependency-check engine
905      * @param project the project being scanned
906      * @param nodes the list of dependency nodes, generally obtained via the
907      * DependencyGraphBuilder
908      * @param buildingRequest the Maven project building request
909      * @param aggregate whether the scan is part of an aggregate build
910      * @return a collection of exceptions that may have occurred while resolving
911      * and scanning the dependencies
912      */
913     private ExceptionCollection collectMavenDependencies(Engine engine, MavenProject project,
914             List<DependencyNode> nodes, ProjectBuildingRequest buildingRequest, boolean aggregate) {
915         ExceptionCollection exCol = null;
916         for (DependencyNode dependencyNode : nodes) {
917             if (artifactScopeExcluded.passes(dependencyNode.getArtifact().getScope())
918                     || artifactTypeExcluded.passes(dependencyNode.getArtifact().getType())) {
919                 continue;
920             }
921 
922             boolean isResolved = false;
923             File artifactFile = null;
924             String artifactId = null;
925             String groupId = null;
926             String version = null;
927             List<ArtifactVersion> availableVersions = null;
928             if (org.apache.maven.artifact.Artifact.SCOPE_SYSTEM.equals(dependencyNode.getArtifact().getScope())) {
929                 for (org.apache.maven.model.Dependency d : project.getDependencies()) {
930                     final Artifact a = dependencyNode.getArtifact();
931                     if (d.getSystemPath() != null && artifactsMatch(d, a)) {
932 
933                         artifactFile = new File(d.getSystemPath());
934                         isResolved = artifactFile.isFile();
935                         groupId = a.getGroupId();
936                         artifactId = a.getArtifactId();
937                         version = a.getVersion();
938                         availableVersions = a.getAvailableVersions();
939                         break;
940                     }
941                 }
942                 if (!isResolved) {
943                     getLog().error("Unable to resolve system scoped dependency: " + dependencyNode.toNodeString());
944                     if (exCol == null) {
945                         exCol = new ExceptionCollection();
946                     }
947                     exCol.addException(new DependencyNotFoundException("Unable to resolve system scoped dependency: "
948                             + dependencyNode.toNodeString()));
949                 }
950             } else {
951                 final ArtifactCoordinate coordinate = TransferUtils.toArtifactCoordinate(dependencyNode.getArtifact());
952                 final Artifact result;
953                 try {
954                     result = artifactResolver.resolveArtifact(buildingRequest, coordinate).getArtifact();
955                 } catch (ArtifactResolverException ex) {
956                     getLog().debug(String.format("Aggregate : %s", aggregate));
957                     boolean addException = true;
958                     if (!aggregate || addReactorDependency(engine, dependencyNode.getArtifact())) {
959                         addException = false;
960                     }
961                     if (addException) {
962                         if (exCol == null) {
963                             exCol = new ExceptionCollection();
964                         }
965                         exCol.addException(ex);
966                     }
967                     continue;
968                 }
969                 if (aggregate && virtualSnapshotsFromReactor
970                         && dependencyNode.getArtifact().isSnapshot() && aggregate
971                         && addSnapshotReactorDependency(engine, dependencyNode.getArtifact())) {
972                     continue;
973                 }
974                 isResolved = result.isResolved();
975                 artifactFile = result.getFile();
976                 groupId = result.getGroupId();
977                 artifactId = result.getArtifactId();
978                 version = result.getVersion();
979                 availableVersions = result.getAvailableVersions();
980             }
981             if (isResolved && artifactFile != null) {
982                 final List<Dependency> deps = engine.scan(artifactFile.getAbsoluteFile(),
983                         project.getName() + ":" + dependencyNode.getArtifact().getScope());
984                 if (deps != null) {
985                     Dependency d = null;
986                     if (deps.size() == 1) {
987                         d = deps.get(0);
988                     } else {
989                         for (Dependency possible : deps) {
990                             if (artifactFile.getAbsoluteFile().equals(possible.getActualFile())) {
991                                 d = possible;
992                                 break;
993                             }
994                         }
995                     }
996                     if (d != null) {
997                         final MavenArtifact ma = new MavenArtifact(groupId, artifactId, version);
998                         d.addAsEvidence("pom", ma, Confidence.HIGHEST);
999                         if (availableVersions != null) {
1000                             for (ArtifactVersion av : availableVersions) {
1001                                 d.addAvailableVersion(av.toString());
1002                             }
1003                         }
1004                         getLog().debug(String.format("Adding project reference %s on dependency %s",
1005                                 project.getName(), d.getDisplayFileName()));
1006                     } else if (getLog().isDebugEnabled()) {
1007                         final String msg = String.format("More than 1 dependency was identified in first pass scan of '%s' in project %s",
1008                                 dependencyNode.getArtifact().getId(), project.getName());
1009                         getLog().debug(msg);
1010                     }
1011                 } else if ("import".equals(dependencyNode.getArtifact().getScope())) {
1012                     final String msg = String.format("Skipping '%s:%s' in project %s as it uses an `import` scope",
1013                             dependencyNode.getArtifact().getId(), dependencyNode.getArtifact().getScope(), project.getName());
1014                     getLog().debug(msg);
1015                 } else {
1016                     final String msg = String.format("No analyzer could be found for '%s:%s' in project %s",
1017                             dependencyNode.getArtifact().getId(), dependencyNode.getArtifact().getScope(), project.getName());
1018                     getLog().warn(msg);
1019                 }
1020             } else {
1021                 final String msg = String.format("Unable to resolve '%s' in project %s",
1022                         dependencyNode.getArtifact().getId(), project.getName());
1023                 getLog().debug(msg);
1024                 if (exCol == null) {
1025                     exCol = new ExceptionCollection();
1026                 }
1027             }
1028         }
1029         return exCol;
1030     }

1032     /**
1033      * Scans the projects dependencies including the default (or defined)
1034      * FileSets.
1035      *
1036      * @param engine the core dependency-check engine
1037      * @param project the project being scanned
1038      * @param nodes the list of dependency nodes, generally obtained via the
1039      * DependencyGraphBuilder
1040      * @param buildingRequest the Maven project building request
1041      * @param aggregate whether the scan is part of an aggregate build
1042      * @return a collection of exceptions that may have occurred while resolving
1043      * and scanning the dependencies
1044      */
1045     private ExceptionCollection collectDependencies(Engine engine, MavenProject project,
1046             List<DependencyNode> nodes, ProjectBuildingRequest buildingRequest, boolean aggregate) {
1047 
1048         ExceptionCollection exCol = null;
1049         exCol = collectMavenDependencies(engine, project, nodes, buildingRequest, aggregate);
1050 
1051         FileSet[] projectScan = null;
1052         if (scanSet == null || scanSet.length == 0) {
1053             // Define the default FileSets
1054             final FileSet resourcesSet = new FileSet();
1055             final FileSet filtersSet = new FileSet();
1056             final FileSet webappSet = new FileSet();
1057             try {
1058                 resourcesSet.setDirectory(new File(project.getBasedir(), "src/main/resources").getCanonicalPath());
1059                 filtersSet.setDirectory(new File(project.getBasedir(), "src/main/filters").getCanonicalPath());
1060                 webappSet.setDirectory(new File(project.getBasedir(), "src/main/webapp").getCanonicalPath());
1061             } catch (IOException ex) {
1062                 if (exCol == null) {
1063                     exCol = new ExceptionCollection();
1064                 }
1065                 exCol.addException(ex);
1066             }
1067             projectScan = new FileSet[]{resourcesSet, filtersSet, webappSet};
1068 
1069         } else if (aggregate) {
1070             projectScan = new FileSet[scanSet.length];
1071             for (int x = 0; x < scanSet.length; x++) {
1072                 //deep copy of the FileSet - modifying the directory if it is not absolute.
1073                 final FileSet copyFrom = scanSet[x];
1074                 final FileSet fsCopy = new FileSet();
1075 
1076                 final File f = new File(copyFrom.getDirectory());
1077                 if (f.isAbsolute()) {
1078                     fsCopy.setDirectory(copyFrom.getDirectory());
1079                 } else {
1080                     try {
1081                         fsCopy.setDirectory(new File(project.getBasedir(), copyFrom.getDirectory()).getCanonicalPath());
1082                     } catch (IOException ex) {
1083                         if (exCol == null) {
1084                             exCol = new ExceptionCollection();
1085                         }
1086                         exCol.addException(ex);
1087                         fsCopy.setDirectory(copyFrom.getDirectory());
1088                     }
1089                 }
1090                 fsCopy.setDirectoryMode(copyFrom.getDirectoryMode());
1091                 fsCopy.setExcludes(copyFrom.getExcludes());
1092                 fsCopy.setFileMode(copyFrom.getFileMode());
1093                 fsCopy.setFollowSymlinks(copyFrom.isFollowSymlinks());
1094                 fsCopy.setIncludes(copyFrom.getIncludes());
1095                 fsCopy.setLineEnding(copyFrom.getLineEnding());
1096                 fsCopy.setMapper(copyFrom.getMapper());
1097                 fsCopy.setModelEncoding(copyFrom.getModelEncoding());
1098                 fsCopy.setOutputDirectory(copyFrom.getOutputDirectory());
1099                 fsCopy.setUseDefaultExcludes(copyFrom.isUseDefaultExcludes());
1100                 projectScan[x] = fsCopy;
1101             }
1102         } else {
1103             projectScan = scanSet;
1104         }
1105 
1106         // Iterate through FileSets and scan included files
1107         final FileSetManager fileSetManager = new FileSetManager();
1108         for (FileSet fileSet : projectScan) {
1109             final String[] includedFiles = fileSetManager.getIncludedFiles(fileSet);
1110             for (String include : includedFiles) {
1111                 final File includeFile = new File(fileSet.getDirectory(), include).getAbsoluteFile();
1112                 if (includeFile.exists()) {
1113                     engine.scan(includeFile, project.getName());
1114                 }
1115                 //TODO - should we add an exception/error reporting for files that do not exist?
1116             }
1117         }
1118 
1119         return exCol;
1120     }

1143      * @return <code>true</code> if the artifact is in the reactor; otherwise
1144      * <code>false</code>

1146     private boolean addVirtualDependencyFromReactor(Engine engine, Artifact artifact, String infoLogTemplate) {
1147 
1148         getLog().debug(String.format("Checking the reactor projects (%d) for %s:%s:%s",
1149                 reactorProjects.size(),
1150                 artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion()));
1151 
1152         for (MavenProject prj : reactorProjects) {
1153 
1154             getLog().debug(String.format("Comparing %s:%s:%s to %s:%s:%s",
1155                     artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion(),
1156                     prj.getGroupId(), prj.getArtifactId(), prj.getVersion()));
1157 
1158             if (prj.getArtifactId().equals(artifact.getArtifactId())
1159                     && prj.getGroupId().equals(artifact.getGroupId())
1160                     && prj.getVersion().equals(artifact.getVersion())) {
1161 
1162                 final String displayName = String.format("%s:%s:%s",
1163                         prj.getGroupId(), prj.getArtifactId(), prj.getVersion());
1164                 getLog().info(String.format(infoLogTemplate,
1165                         displayName));
1166                 final File pom = new File(prj.getBasedir(), "pom.xml");
1167                 final Dependency d;
1168                 if (pom.isFile()) {
1169                     getLog().debug("Adding virtual dependency from pom.xml");
1170                     d = new Dependency(pom, true);
1171                 } else {
1172                     d = new Dependency(true);
1173                 }
1174                 final String key = String.format("%s:%s:%s", artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion());
1175                 d.setSha1sum(Checksum.getSHA1Checksum(key));
1176                 d.setSha256sum(Checksum.getSHA256Checksum(key));
1177                 d.setMd5sum(Checksum.getMD5Checksum(key));
1178                 d.setEcosystem(JarAnalyzer.DEPENDENCY_ECOSYSTEM);
1179                 d.setDisplayFileName(displayName);
1180 
1181                 d.addEvidence(EvidenceType.PRODUCT, "project", "artifactid", prj.getArtifactId(), Confidence.HIGHEST);
1182                 d.addEvidence(EvidenceType.VENDOR, "project", "artifactid", prj.getArtifactId(), Confidence.LOW);
1183 
1184                 d.addEvidence(EvidenceType.VENDOR, "project", "groupid", prj.getGroupId(), Confidence.HIGHEST);
1185                 d.addEvidence(EvidenceType.PRODUCT, "project", "groupid", prj.getGroupId(), Confidence.LOW);
1186                 d.setEcosystem(JarAnalyzer.DEPENDENCY_ECOSYSTEM);
1187                 final Identifier id = new Identifier();
1188                 id.setType("maven");
1189                 id.setConfidence(Confidence.HIGHEST);
1190                 id.setValue(displayName);
1191                 d.addIdentifier(id);
1192                 //TODO unify the setName/version and package path - they are equivelent ideas submitted by two seperate committers
1193                 d.setName(String.format("%s:%s", prj.getGroupId(), prj.getArtifactId()));
1194                 d.setVersion(prj.getVersion());
1195                 d.setPackagePath(displayName);
1196                 if (prj.getDescription() != null) {
1197                     JarAnalyzer.addDescription(d, prj.getDescription(), "project", "description");
1198                 }
1199                 for (License l : prj.getLicenses()) {
1200                     final StringBuilder license = new StringBuilder();
1201                     if (l.getName() != null) {
1202                         license.append(l.getName());
1203                     }
1204                     if (l.getUrl() != null) {
1205                         license.append(" ").append(l.getUrl());
1206                     }
1207                     if (d.getLicense() == null) {
1208                         d.setLicense(license.toString());
1209                     } else if (!d.getLicense().contains(license)) {
1210                         d.setLicense(String.format("%s%n%s", d.getLicense(), license.toString()));
1211                     }
1212                 }
1213                 engine.addDependency(d);
1214                 return true;
1215             }
1216         }

1254     protected void runCheck() throws MojoExecutionException, MojoFailureException {
1255         try (Engine engine = initializeEngine()) {
1256             ExceptionCollection exCol = scanDependencies(engine);
1257             try {
1258                 engine.analyzeDependencies();
1259             } catch (ExceptionCollection ex) {
1260                 exCol = handleAnalysisExceptions(exCol, ex);
1261             }
1262             if (exCol == null || !exCol.isFatal()) {
1263

1265                 if (outputDir == null) {

1269                 }

1278                     }

1280                         throw new MojoExecutionException("One or more exceptions occurred during dependency-check analysis", exCol);
1281                     } else {
1282                         getLog().debug("Error writing the report", ex);
1283                     }
1284                 }
1285                 showSummary(this.getProject(), engine.getDependencies());
1286                 checkForFailure(engine.getDependencies());
1287                 if (exCol != null && this.isFailOnError()) {
1288                     throw new MojoExecutionException("One or more exceptions occurred during dependency-check analysis", exCol);
1289                 }
1290             }
1291         } catch (DatabaseException ex) {

1293                 getLog().debug("Database connection error", ex);

1296             if (this.isFailOnError()) {

1305     /**
1306      * Combines the two exception collections and if either are fatal, throw an
1307      * MojoExecutionException
1308      *
1309      * @param currentEx the primary exception collection
1310      * @param newEx the new exception collection to add
1311      * @return the combined exception collection
1312      * @throws MojoExecutionException thrown if dependency-check is configured
1313      * to fail on errors
1314      */
1315     private ExceptionCollection handleAnalysisExceptions(ExceptionCollection currentEx, ExceptionCollection newEx) throws MojoExecutionException {
1316         ExceptionCollection returnEx = currentEx;
1317         if (returnEx == null) {
1318             returnEx = newEx;
1319         } else {
1320             returnEx.getExceptions().addAll(newEx.getExceptions());
1321             if (newEx.isFatal()) {
1322                 returnEx.setFatal(true);
1323             }
1324         }
1325         if (returnEx.isFatal()) {
1326             final String msg = String.format("Fatal exception(s) analyzing %s", getProject().getName());
1327             if (this.isFailOnError()) {
1328                 throw new MojoExecutionException(msg, returnEx);
1329             }
1330             getLog().error(msg);
1331             if (getLog().isDebugEnabled()) {
1332                 getLog().debug(returnEx);
1333             }
1334         } else {
1335             final String msg = String.format("Exception(s) analyzing %s", getProject().getName());
1336             if (getLog().isDebugEnabled()) {
1337                 getLog().debug(msg, returnEx);
1338             }
1339         }
1340         return returnEx;

1714      * Checks to see if a vulnerability has been identified with a CVSS score
1715      * that is above the threshold set in the configuration.

1717      * @param dependencies the list of dependency objects

1722         final StringBuilder ids = new StringBuilder();
1723         for (Dependency d : dependencies) {
1724             boolean addName = true;
1725             for (Vulnerability v : d.getVulnerabilities()) {

1727                     if (addName) {
1728                         addName = false;
1729                         ids.append(NEW_LINE).append(d.getFileName()).append(": ");
1730                         ids.append(v.getName());
1731                     } else {
1732                         ids.append(", ").append(v.getName());
1733                     }
1734                 }
1735             }
1736         }

1743                 } else {
1744                     msg = String.format("%n%nOne or more dependencies were identified with vulnerabilities that have a CVSS score greater than or "
1745                             + "equal to '%.1f': %n%s%n%nSee the dependency-check report for more details.%n%n", failBuildOnCVSS, ids.toString());
1746                 }
1747             } else {
