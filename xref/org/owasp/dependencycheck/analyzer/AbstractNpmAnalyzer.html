40  /**
41   * An abstract NPM analyzer that contains common methods for concrete
42   * implementations.
43   *
44   * @author Steve Springett
45   */
46  @ThreadSafe
47  public abstract class AbstractNpmAnalyzer extends AbstractFileTypeAnalyzer {
48  
49      /**
50       * The logger.
51       */
52      private static final Logger LOGGER = LoggerFactory.getLogger(AbstractNpmAnalyzer.class);
53  
54      /**
55       * A descriptor for the type of dependencies processed or added by this
56       * analyzer.
57       */
58      public static final String NPM_DEPENDENCY_ECOSYSTEM = "npm";
59      /**
60       * The file name to scan.
61       */
62      private static final String PACKAGE_JSON = "package.json";
63  
64      /**
65       * Determines if the file can be analyzed by the analyzer.
66       *
67       * @param pathname the path to the file
68       * @return true if the file can be analyzed by the given analyzer; otherwise
69       * false
70       */
71      @Override
72      public boolean accept(File pathname) {
73          boolean accept = super.accept(pathname);
74          if (accept) {
75              try {
76                  accept |= shouldProcess(pathname);
77              } catch (AnalysisException ex) {
78                  throw new RuntimeException(ex.getMessage(), ex.getCause());
79              }
80          }
81  
82          return accept;
83      }
84  
85      /**
86       * Determines if the path contains "/node_modules/" (i.e. it is a child
87       * module. This analyzer does not scan child modules.
88       *
89       * @param pathname the path to test
90       * @return <code>true</code> if the path does not contain "/node_modules/"
91       * @throws AnalysisException thrown if the canonical path cannot be obtained
92       * from the given file
93       */
94      protected boolean shouldProcess(File pathname) throws AnalysisException {
95          try {
96              // Do not scan the node_modules directory
97              if (pathname.getCanonicalPath().contains(File.separator + "node_modules" + File.separator)) {
98                  LOGGER.debug("Skipping analysis of node module: " + pathname.getCanonicalPath());
99                  return false;
100             }
101         } catch (IOException ex) {
102             throw new AnalysisException("Unable to process dependency", ex);
103         }
104         return true;
105     }
106 
107     /**
108      * Construct a dependency object.
109      *
110      * @param dependency the parent dependency
111      * @param name the name of the dependency to create
112      * @param version the version of the dependency to create
113      * @param scope the scope of the dependency being created
114      * @return the generated dependency
115      */
116     protected Dependency createDependency(Dependency dependency, String name, String version, String scope) {
117         final Dependency nodeModule = new Dependency(new File(dependency.getActualFile() + "?" + name), true);
118         nodeModule.setEcosystem(NPM_DEPENDENCY_ECOSYSTEM);
119         //this is virtual - the sha1 is purely for the hyperlink in the final html report
120         nodeModule.setSha1sum(Checksum.getSHA1Checksum(String.format("%s:%s", name, version)));
121         nodeModule.setSha256sum(Checksum.getSHA256Checksum(String.format("%s:%s", name, version)));
122         nodeModule.setMd5sum(Checksum.getMD5Checksum(String.format("%s:%s", name, version)));
123         nodeModule.addEvidence(EvidenceType.PRODUCT, "package.json", "name", name, Confidence.HIGHEST);
124         nodeModule.addEvidence(EvidenceType.VENDOR, "package.json", "name", name, Confidence.HIGH);
125         nodeModule.addEvidence(EvidenceType.VERSION, "package.json", "version", version, Confidence.HIGHEST);
126         nodeModule.addProjectReference(dependency.getName() + ": " + scope);
127         nodeModule.setName(name);
128         nodeModule.setVersion(version);
129         nodeModule.addIdentifier("npm", String.format("%s:%s", name, version), null, Confidence.HIGHEST);
130         return nodeModule;
131     }
132 
133     /**
134      * Processes a part of package.json (as defined by JsonArray) and update the
135      * specified dependency with relevant info.
136      *
137      * @param engine the dependency-check engine
138      * @param dependency the Dependency to update
139      * @param jsonArray the jsonArray to parse
140      * @param depType the dependency type
141      */
142     protected void processPackage(Engine engine, Dependency dependency, JsonArray jsonArray, String depType) {
143         final JsonObjectBuilder builder = Json.createObjectBuilder();
144         for (JsonString str : jsonArray.getValuesAs(JsonString.class)) {
145             builder.add(str.toString(), "");
146         }
147         final JsonObject jsonObject = builder.build();
148         processPackage(engine, dependency, jsonObject, depType);
149     }
150 
151     /**
152      * Processes a part of package.json (as defined by JsonObject) and update
153      * the specified dependency with relevant info.
154      *
155      * @param engine the dependency-check engine
156      * @param dependency the Dependency to update
157      * @param jsonObject the jsonObject to parse
158      * @param depType the dependency type
159      */
160     protected void processPackage(Engine engine, Dependency dependency, JsonObject jsonObject, String depType) {
161         for (int i = 0; i < jsonObject.size(); i++) {
162             for (Map.Entry<String, JsonValue> entry : jsonObject.entrySet()) {
163 
164                 final String name = entry.getKey();
165                 String version = "";
166                 if (entry.getValue() != null && entry.getValue().getValueType() == JsonValue.ValueType.STRING) {
167                     version = ((JsonString) entry.getValue()).getString();
168                 }
169                 final Dependency existing = findDependency(engine, name, version);
170                 if (existing == null) {
171                     final Dependency nodeModule = createDependency(dependency, name, version, depType);
172                     engine.addDependency(nodeModule);
173                 } else {
174                     existing.addProjectReference(dependency.getName() + ": " + depType);
175                 }
176             }
177         }
178     }

180     /**
181      * Adds information to an evidence collection from the node json
182      * configuration.
183      *
184      * @param dep the dependency to add the evidence
185      * @param t the type of evidence to add
186      * @param json information from node.js
187      * @return the actual string set into evidence
188      * @param key the key to obtain the data from the json information
189      */
190     private static String addToEvidence(Dependency dep, EvidenceType t, JsonObject json, String key) {
191         String evidenceStr = null;
192         if (json.containsKey(key)) {
193             final JsonValue value = json.get(key);
194             if (value instanceof JsonString) {
195                 evidenceStr = ((JsonString) value).getString();
196                 dep.addEvidence(t, PACKAGE_JSON, key, evidenceStr, Confidence.HIGHEST);
197             } else if (value instanceof JsonObject) {
198                 final JsonObject jsonObject = (JsonObject) value;
199                 for (final Map.Entry<String, JsonValue> entry : jsonObject.entrySet()) {
200                     final String property = entry.getKey();
201                     final JsonValue subValue = entry.getValue();
202                     if (subValue instanceof JsonString) {
203                         evidenceStr = ((JsonString) subValue).getString();
204                         dep.addEvidence(t, PACKAGE_JSON,
205                                 String.format("%s.%s", key, property),
206                                 evidenceStr,
207                                 Confidence.HIGHEST);
208                     } else {
209                         LOGGER.warn("JSON sub-value not string as expected: {}", subValue);
210                     }
211                 }
212             } else {
213                 LOGGER.warn("JSON value not string or JSON object as expected: {}", value);
214             }
215         }
216         return evidenceStr;
217     }

219     /**
220      * Locates the dependency from the list of dependencies that have been
221      * scanned by the engine.
222      *
223      * @param engine the dependency-check engine
224      * @param name the name of the dependency to find
225      * @param version the version of the dependency to find
226      * @return the identified dependency; otherwise null
227      */
228     protected Dependency findDependency(Engine engine, String name, String version) {
229         for (Dependency d : engine.getDependencies()) {
230             if (NPM_DEPENDENCY_ECOSYSTEM.equals(d.getEcosystem()) && name.equals(d.getName()) && version != null && d.getVersion() != null) {
231                 final String dependencyVersion = d.getVersion();
232                 if (DependencyBundlingAnalyzer.npmVersionsMatch(version, dependencyVersion)) {
233                     return d;
234                 }
235             }
236         }
237         return null;
238     }

240     /**
241      * Collects evidence from the given JSON for the associated dependency.
242      *
243      * @param json the JSON that contains the evidence to collect
244      * @param dependency the dependency to add the evidence too
245      */
246     public void gatherEvidence(final JsonObject json, Dependency dependency) {
247         String displayName = null;
248         if (json.containsKey("name")) {
249             final Object value = json.get("name");
250             if (value instanceof JsonString) {
251                 final String valueString = ((JsonString) value).getString();
252                 displayName = valueString;
253                 dependency.setName(valueString);
254                 dependency.setPackagePath(valueString);
255                 dependency.addEvidence(EvidenceType.PRODUCT, PACKAGE_JSON, "name", valueString, Confidence.HIGHEST);
256                 dependency.addEvidence(EvidenceType.VENDOR, PACKAGE_JSON, "name", valueString, Confidence.HIGH);
257             } else {
258                 LOGGER.warn("JSON value not string as expected: {}", value);
259             }
260         }
261         final String desc = addToEvidence(dependency, EvidenceType.PRODUCT, json, "description");
262         dependency.setDescription(desc);
263         final String vendor = addToEvidence(dependency, EvidenceType.VENDOR, json, "author");
264         final String version = addToEvidence(dependency, EvidenceType.VERSION, json, "version");
265         if (version != null) {
266             displayName = String.format("%s:%s", displayName, version);
267             dependency.setVersion(version);
268             dependency.addIdentifier("npm", String.format("%s:%s", dependency.getName(), version), null, Confidence.HIGHEST);
269         }
270         if (displayName != null) {
271             dependency.setDisplayFileName(displayName);
272             dependency.setPackagePath(displayName);
273         } else {
274             LOGGER.warn("Unable to determine package name or version for {}", dependency.getActualFilePath());
275             if (vendor != null && !vendor.isEmpty()) {
276                 dependency.setDisplayFileName(String.format("%s package.json", vendor));
277             }
278         }
279         // Adds the license if defined in package.json
280         if (json.containsKey("license")) {
281             final Object value = json.get("license");
282             if (value instanceof JsonString) {
283                 dependency.setLicense(json.getString("license"));
284             } else if (value instanceof JsonArray) {
285                 final JsonArray array = (JsonArray) value;
286                 final StringBuilder sb = new StringBuilder();
287                 boolean addComma = false;
288                 for (int x = 0; x < array.size(); x++) {
289                     if (!array.isNull(x)) {
290                         if (addComma) {
291                             sb.append(", ");
292                         } else {
293                             addComma = true;
294                         }
295                         if (ValueType.STRING == array.get(x).getValueType()) {
296                             sb.append(array.getString(x));
297                         } else {
298                             final JsonObject lo = array.getJsonObject(x);
299                             if (lo.containsKey("type") && !lo.isNull("type")
300                                     && lo.containsKey("url") && !lo.isNull("url")) {
301                                 final String license = String.format("%s (%s)", lo.getString("type"), lo.getString("url"));
302                                 sb.append(license);
303                             } else if (lo.containsKey("type") && !lo.isNull("type")) {
304                                 sb.append(lo.getString("type"));
305                             } else if (lo.containsKey("url") && !lo.isNull("url")) {
306                                 sb.append(lo.getString("url"));
307                             }
308                         }
309                     }
310                 }
311                 dependency.setLicense(sb.toString());
312             } else {
313                 dependency.setLicense(json.getJsonObject("license").getString("type"));
314             }
315         }
316     }
