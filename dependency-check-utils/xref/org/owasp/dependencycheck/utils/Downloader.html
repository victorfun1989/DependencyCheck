38   * A utility to download files from the Internet.

45       * The logger.
46       */

48      /**
49       * The maximum number of redirects that will be followed when attempting to
50       * download a file.
51       */
52      private static final int MAX_REDIRECT_ATTEMPTS = 5;

55       * The default HTTP request method for query timestamp

84      /**
85       * Retrieves a file from a given URL and saves it to the outputPath.
86       *
87       * @param url the URL of the file to download
88       * @param outputPath the path to the save the file to
89       * @throws DownloadFailedException is thrown if there is an error
90       * downloading the file
91       */
92      public void fetchFile(URL url, File outputPath) throws DownloadFailedException {
93          fetchFile(url, outputPath, true);
94      }
95  
96      /**
97       * Retrieves a file from a given URL and saves it to the outputPath.
98       *
99       * @param url the URL of the file to download
100      * @param outputPath the path to the save the file to
101      * @param useProxy whether to use the configured proxy when downloading
102      * files
103      * @throws DownloadFailedException is thrown if there is an error
104      * downloading the file
105      */
106     public void fetchFile(URL url, File outputPath, boolean useProxy) throws DownloadFailedException {
107         if ("file".equalsIgnoreCase(url.getProtocol())) {
108             final File file;
109             try {
110                 file = new File(url.toURI());
111             } catch (URISyntaxException ex) {
112                 final String msg = format("Download failed, unable to locate '%s'", url.toString());
113                 throw new DownloadFailedException(msg);
114             }
115             if (file.exists()) {
116                 try {
117                     org.apache.commons.io.FileUtils.copyFile(file, outputPath);
118                 } catch (IOException ex) {
119                     final String msg = format("Download failed, unable to copy '%s' to '%s'", url.toString(), outputPath.getAbsolutePath());
120                     throw new DownloadFailedException(msg, ex);
121                 }
122             } else {
123                 final String msg = format("Download failed, file ('%s') does not exist", url.toString());
124                 throw new DownloadFailedException(msg);
125             }
126         } else {
127             HttpURLConnection conn = null;
128             try {
129                 LOGGER.debug("Attempting download of {}", url.toString());
130                 conn = connFactory.createHttpURLConnection(url, useProxy);
131                 conn.setRequestProperty("Accept-Encoding", "gzip, deflate");
132                 conn.connect();
133                 int status = conn.getResponseCode();
134                 int redirectCount = 0;
135                 while ((status == HttpURLConnection.HTTP_MOVED_TEMP
136                         || status == HttpURLConnection.HTTP_MOVED_PERM
137                         || status == HttpURLConnection.HTTP_SEE_OTHER)
138                         && MAX_REDIRECT_ATTEMPTS > redirectCount++) {
139                     final String location = conn.getHeaderField("Location");
140                     try {
141                         conn.disconnect();
142                     } finally {
143                         conn = null;
144                     }
145                     LOGGER.debug("Download is being redirected from {} to {}", url.toString(), location);
146                     conn = connFactory.createHttpURLConnection(new URL(location), useProxy);
147                     conn.setRequestProperty("Accept-Encoding", "gzip, deflate");
148                     conn.connect();
149                     status = conn.getResponseCode();
150                 }
151                 if (status != 200) {
152                     try {
153                         conn.disconnect();
154                     } finally {
155                         conn = null;
156                     }
157                     final String msg = format("Error downloading file %s; received response code %s.", url.toString(), status);
158                     throw new DownloadFailedException(msg);
159 
160                 }
161             } catch (IOException ex) {
162                 try {
163                     if (conn != null) {
164                         conn.disconnect();
165                     }
166                 } finally {
167                     conn = null;
168                 }
169                 if ("Connection reset".equalsIgnoreCase(ex.getMessage())) {
170                     final String msg = format("TLS Connection Reset%nPlease see "
171                             + "http://jeremylong.github.io/DependencyCheck/data/tlsfailure.html "
172                             + "for more information regarding how to resolve the issue.");
173                     LOGGER.error(msg);
174                     throw new DownloadFailedException(msg, ex);
175                 }
176                 final String msg = format("Error downloading file %s; unable to connect.", url.toString());
177                 throw new DownloadFailedException(msg, ex);
178             }
179 
180             final String encoding = conn.getContentEncoding();
181             InputStream reader = null;
182             try (OutputStream out = new FileOutputStream(outputPath);
183                     BufferedOutputStream writer = new BufferedOutputStream(out)) {
184                 if (encoding != null && "gzip".equalsIgnoreCase(encoding)) {
185                     reader = new GZIPInputStream(conn.getInputStream());
186                 } else if (encoding != null && "deflate".equalsIgnoreCase(encoding)) {
187                     reader = new InflaterInputStream(conn.getInputStream());
188                 } else {
189                     reader = conn.getInputStream();
190                 }
191 
192                 final byte[] buffer = new byte[4096];
193                 int bytesRead;
194                 while ((bytesRead = reader.read(buffer)) > 0) {
195                     writer.write(buffer, 0, bytesRead);
196                 }
197                 LOGGER.debug("Download of {} complete", url.toString());
198             } catch (IOException ex) {
199                 checkForCommonExceptionTypes(ex);
200                 final String msg = format("Error saving '%s' to file '%s'%nConnection Timeout: %d%nEncoding: %s%n",
201                         url.toString(), outputPath.getAbsolutePath(), conn.getConnectTimeout(), encoding);
202                 throw new DownloadFailedException(msg, ex);
203             } catch (Exception ex) {
204                 final String msg = format("Unexpected exception saving '%s' to file '%s'%nConnection Timeout: %d%nEncoding: %s%n",
205                         url.toString(), outputPath.getAbsolutePath(), conn.getConnectTimeout(), encoding);
206                 throw new DownloadFailedException(msg, ex);
207             } finally {
208                 if (reader != null) {
209                     try {
210                         reader.close();
211                     } catch (IOException ex) {
212                         LOGGER.trace("Error closing the reader in Downloader.", ex);
213                     }
214                 }
215                 try {
216                     conn.disconnect();
217                 } finally {
218                     conn = null;
219                 }
220             }
221         }
222     }

224     /**
225      * Makes an HTTP Head request to retrieve the last modified date of the
226      * given URL. If the file:// protocol is specified, then the lastTimestamp
227      * of the file is returned.
228      *
229      * @param url the URL to retrieve the timestamp from
230      * @return an epoch timestamp
231      * @throws DownloadFailedException is thrown if an exception occurs making
232      * the HTTP request
233      */
234     public long getLastModified(URL url) throws DownloadFailedException {
235         return getLastModified(url, false);
236     }
237 
238     /**
239      * Makes an HTTP Head request to retrieve the last modified date of the
240      * given URL. If the file:// protocol is specified, then the lastTimestamp
241      * of the file is returned.
242      *
243      * @param url the URL to retrieve the timestamp from
244      * @param isRetry indicates if this is a retry - to prevent endless loop and
245      * stack overflow
246      * @return an epoch timestamp
247      * @throws DownloadFailedException is thrown if an exception occurs making
248      * the HTTP request
249      */
250     private long getLastModified(URL url, boolean isRetry) throws DownloadFailedException {
251         long timestamp = 0;
252         //TODO add the FTP protocol?
253         if ("file".equalsIgnoreCase(url.getProtocol())) {
254             final File lastModifiedFile;
255             try {
256                 lastModifiedFile = new File(url.toURI());
257             } catch (URISyntaxException ex) {
258                 final String msg = format("Unable to locate '%s'", url.toString());
259                 throw new DownloadFailedException(msg, ex);
260             }
261             timestamp = lastModifiedFile.lastModified();
262         } else {
263             final String httpMethod = determineHttpMethod();
264             HttpURLConnection conn = null;
265             try {
266                 conn = connFactory.createHttpURLConnection(url);
267                 conn.setRequestMethod(httpMethod);
268                 conn.connect();
269                 final int t = conn.getResponseCode();
270                 if (t >= 200 && t < 300) {
271                     timestamp = conn.getLastModified();
272                 } else {
273                     throw new DownloadFailedException(format("%s request returned a non-200 status code: %s", httpMethod, url));
274                 }
275             } catch (URLConnectionFailureException ex) {
276                 throw new DownloadFailedException(format("Error creating URL Connection for HTTP %s request: %s", httpMethod, url), ex);
277             } catch (IOException ex) {
278                 checkForCommonExceptionTypes(ex);
279                 LOGGER.error(String.format("IO Exception connecting to %s: %s", url, ex.getMessage()));
280                 LOGGER.debug("Exception details", ex);
281                 if (ex.getCause() != null) {
282                     LOGGER.debug("IO Exception cause: " + ex.getCause().getMessage(), ex.getCause());
283                 }
284                 try {
285                     //retry
286                     if (!isRetry && settings.getBoolean(Settings.KEYS.DOWNLOADER_QUICK_QUERY_TIMESTAMP)) {
287                         settings.setBoolean(Settings.KEYS.DOWNLOADER_QUICK_QUERY_TIMESTAMP, false);
288                         return getLastModified(url, true);
289                     }
290                 } catch (InvalidSettingException ex1) {
291                     LOGGER.debug("invalid setting?", ex1);
292                 }
293                 throw new DownloadFailedException(format("Error making HTTP %s request to %s", httpMethod, url), ex);
294             } finally {
295                 if (conn != null) {
296                     try {
297                         conn.disconnect();
298                     } finally {
299                         conn = null;
300                     }
301                 }
302             }
303         }
304         return timestamp;
305     }

307     /**
308      * Analyzes the IOException, logs the appropriate information for debugging
309      * purposes, and then throws a DownloadFailedException that wraps the IO
310      * Exception for common IO Exceptions. This is to provide additional details
311      * to assist in resolution of the exception.
312      *
313      * @param ex the original exception
314      * @throws DownloadFailedException a wrapper exception that contains the
315      * original exception as the cause
316      */
317     protected synchronized void checkForCommonExceptionTypes(IOException ex) throws DownloadFailedException {
318         Throwable cause = ex;
319         while (cause != null) {
320             if (cause instanceof java.net.UnknownHostException) {
321                 final String msg = format("Unable to resolve domain '%s'", cause.getMessage());
322                 LOGGER.error(msg);
323                 throw new DownloadFailedException(msg, cause);
324             }
325             if (cause instanceof InvalidAlgorithmParameterException) {
326                 final String keystore = System.getProperty("javax.net.ssl.keyStore");
327                 final String version = System.getProperty("java.version");
328                 final String vendor = System.getProperty("java.vendor");
329                 LOGGER.info("Error making HTTPS request - InvalidAlgorithmParameterException");
330                 LOGGER.info("There appears to be an issue with the installation of Java and the cacerts."
331                         + "See closed issue #177 here: https://github.com/jeremylong/DependencyCheck/issues/177");
332                 LOGGER.info("Java Info:\njavax.net.ssl.keyStore='{}'\njava.version='{}'\njava.vendor='{}'",
333                         keystore, version, vendor);
334                 throw new DownloadFailedException("Error making HTTPS request. Please see the log for more details.");
335             }
336             cause = cause.getCause();
337         }
338     }

340     /**
341      * Returns the HEAD or GET HTTP method. HEAD is the default.
342      *
343      * @return the HTTP method to use
344      */
345     private String determineHttpMethod() {
346         return isQuickQuery() ? HEAD : GET;
347     }
348 
349     /**
350      * Determines if the HTTP method GET or HEAD should be used to check the
351      * timestamp on external resources.
352      *
353      * @return true if configured to use HEAD requests
354      */
355     private boolean isQuickQuery() {
356         boolean quickQuery;
357 
358         try {
359             quickQuery = settings.getBoolean(Settings.KEYS.DOWNLOADER_QUICK_QUERY_TIMESTAMP, true);
360         } catch (InvalidSettingException e) {
361             if (LOGGER.isTraceEnabled()) {
362                 LOGGER.trace("Invalid settings : {}", e.getMessage(), e);
363             }
364             quickQuery = true;
365         }
366         return quickQuery;
367     }
